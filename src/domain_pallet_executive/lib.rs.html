113    impl<T: Config> Pallet<T> {
114        /// Sets new runtime code after doing necessary checks.
115        /// Same as frame_system::Call::set_code but without root origin.
116        #[pallet::call_index(0)]
117        #[pallet::weight((T::WeightInfo::set_code(), DispatchClass::Mandatory))]
118        pub fn set_code(origin: OriginFor<T>, code: Vec<u8>) -> DispatchResult {
119            ensure_none(origin)?;
120            // we dont need to check version here since the version check already happens
121            // on the consensus side before issuing a domain runtime upgrade.
122            <frame_system::pallet::Pallet<T>>::can_set_code(&code, false).into_result()?;
123            <T as frame_system::Config>::OnSetCode::set_code(code)?;
124            Ok(())
125        }

128    #[pallet::inherent]
129    impl<T: Config> ProvideInherent for Pallet<T> {
130        type Call = Call<T>;
131        type Error = InherentError;
132        const INHERENT_IDENTIFIER: InherentIdentifier = INHERENT_IDENTIFIER;
133
134        fn create_inherent(data: &InherentData) -> Option<Self::Call> {
135            let inherent_data = data
136                .get_data::<InherentType>(&INHERENT_IDENTIFIER)
137                .expect("Executive inherent data not correctly encoded")
138                .expect("Executive inherent data must be provided");
139
140            inherent_data.maybe_code.map(|code| Call::set_code { code })
141        }
142
143        fn is_inherent_required(data: &InherentData) -> Result<Option<Self::Error>, Self::Error> {
144            let inherent_data = data
145                .get_data::<InherentType>(&INHERENT_IDENTIFIER)
146                .expect("Executive inherent data not correctly encoded")
147                .expect("Executive inherent data must be provided");
148
149            Ok(if inherent_data.maybe_code.is_none() {
150                None
151            } else {
152                Some(InherentError::MissingRuntimeCode)
153            })
154        }
155
156        fn check_inherent(call: &Self::Call, data: &InherentData) -> Result<(), Self::Error> {
157            let inherent_data = data
158                .get_data::<InherentType>(&INHERENT_IDENTIFIER)
159                .expect("Executive inherent data not correctly encoded")
160                .expect("Executive inherent data must be provided");
161
162            if let Some(provided_code) = inherent_data.maybe_code {
163                if let Call::set_code { code } = call
164                    && code != &provided_code
165                {
166                    return Err(InherentError::IncorrectRuntimeCode);
167                }
168            } else {
169                return Err(InherentError::MissingRuntimeCode);
170            }
171
172            Ok(())
173        }
174
175        fn is_inherent(call: &Self::Call) -> bool {
176            matches!(call, Call::set_code { .. })
177        }
178    }

303    // TODO: https://github.com/paritytech/substrate/issues/10711
304    fn initial_checks(block: &BlockOf<ExecutiveConfig>) -> u32 {
305        sp_tracing::enter_span!(sp_tracing::Level::TRACE, "initial_checks");
306        let header = block.header();
307
308        // Check that `parent_hash` is correct.
309        let n = *header.number();
310        assert!(
311            n > BlockNumberFor::<ExecutiveConfig>::zero()
312                && <frame_system::Pallet<ExecutiveConfig>>::block_hash(
313                    n - BlockNumberFor::<ExecutiveConfig>::one()
314                ) == *header.parent_hash(),
315            "Parent hash should be valid.",
316        );
317
318        match ExecutiveConfig::ensure_inherents_are_first(block) {
319            Ok(num) => num,
320            Err(i) => panic!("Invalid inherent position for extrinsic at index {i}"),
321        }
322    }

324    /// Wrapped `frame_executive::Executive::execute_block`.
325    ///
326    /// The purpose is to use our custom [`Executive::apply_extrinsic`] and
327    /// the [`Executive::finalize_block`] logic.
328    pub fn execute_block(block: BlockOf<ExecutiveConfig>) {
329        sp_io::init_tracing();
330        sp_tracing::within_span! {
331            sp_tracing::info_span!("execute_block", ?block);
332            // Execute `on_runtime_upgrade` and `on_initialize`.
333            let mode = Self::initialize_block(block.header());
334            let num_inherents = Self::initial_checks(&block) as usize;
335            let (header, extrinsics) = block.deconstruct();
336            let num_extrinsics = extrinsics.len();
337
338            if mode == ExtrinsicInclusionMode::OnlyInherents && num_extrinsics > num_inherents {
339                // Invalid block
340                panic!("Only inherents are allowed in this block")
341            }
342
343            Self::apply_extrinsics(extrinsics.into_iter());
344
345            // In this case there were no transactions to trigger this state transition:
346            if !<frame_system::Pallet<ExecutiveConfig>>::inherents_applied() {
347                defensive_assert!(num_inherents == num_extrinsics);
348                frame_executive::Executive::<
349                    ExecutiveConfig,
350                    BlockOf<ExecutiveConfig>,
351                    Context,
352                    UnsignedValidator,
353                    AllPalletsWithSystem,
354                    COnRuntimeUpgrade,
355                >::inherents_applied();
356            }
357
358            <frame_system::Pallet<ExecutiveConfig>>::note_finished_extrinsics();
359            <ExecutiveConfig as frame_system::Config>::PostTransactions::post_transactions();
360
361            Self::idle_and_finalize_hook(*header.number());
362            Self::final_checks(&header);
363        }
364    }

366    /// Wrapped `frame_executive::Executive::apply_extrinsics`.
367    fn apply_extrinsics(extrinsics: impl Iterator<Item = ExtrinsicOf<ExecutiveConfig>>) {
368        extrinsics.into_iter().for_each(|e| {
369            if let Err(e) = Self::apply_extrinsic(e) {
370                let err: &'static str = e.into();
371                panic!("{}", err)
372            }
373        });
374    }

388    /// Wrapped `frame_executive::Executive::on_idle_hook` and `frame_executive::Executive::on_finalize_hook`
389    fn idle_and_finalize_hook(block_number: NumberFor<BlockOf<ExecutiveConfig>>) {
390        let weight = <frame_system::Pallet<ExecutiveConfig>>::block_weight();
391        let max_weight = <<ExecutiveConfig as frame_system::Config>::BlockWeights as frame_support::traits::Get<_>>::get().max_block;
392        let remaining_weight = max_weight.saturating_sub(weight.total());
393
394        if remaining_weight.all_gt(Weight::zero()) {
395            let used_weight = AllPalletsWithSystem::on_idle(block_number, remaining_weight);
396            <frame_system::Pallet<ExecutiveConfig>>::register_extra_weight_unchecked(
397                used_weight,
398                DispatchClass::Mandatory,
399            );
400        }
401
402        AllPalletsWithSystem::on_finalize(block_number);
403    }

405    /// Wrapped `frame_executive::Executive::apply_extrinsic`.
406    ///
407    /// Note the storage root in the beginning.
408    pub fn apply_extrinsic(uxt: ExtrinsicOf<ExecutiveConfig>) -> ApplyExtrinsicResult {
409        // apply the extrinsic within another transaction so that changes can be reverted.
410        let res = with_storage_layer(|| {
411            frame_executive::Executive::<
412                ExecutiveConfig,
413                BlockOf<ExecutiveConfig>,
414                Context,
415                UnsignedValidator,
416                AllPalletsWithSystem,
417                COnRuntimeUpgrade,
418            >::apply_extrinsic(uxt.clone())
419            .map_err(|err| DispatchError::Other(err.into()))
420        });
421
422        // apply extrinsic failed with transaction validity error
423        // this could happen for following scenarios
424        // - Bad extrinsic Signature
425        //      This extrinsic will be ignored by the operators during the bundle check
426        //      and marks such bundle as Invalid.
427        // - Extrinsic execution failed
428        //      There are multiple scenarios why this can happen
429        //      - Inherent extrinsic failed. If this happens, we should fail to apply extrinsic
430        //      - Pre and Post dispatch fails. Check the test `test_domain_block_builder_include_ext_with_failed_predispatch`
431        //        why this could happen. If it fail due to this, then we revert the inner storage changes
432        //        but still include extrinsic so that we can clear inconsistency between block body and trace roots.
433        match res {
434            Ok(dispatch_outcome) => Ok(dispatch_outcome),
435            Err(err) => {
436                let encoded = uxt.encode();
437                let (maybe_signer, dispatch_info) =
438                    ExecutiveConfig::ExtrinsicStorageFees::extract_signer(uxt);
439                // if this is mandatory extrinsic, then transaction should not execute
440                // we should fail here.
441                if dispatch_info.class == DispatchClass::Mandatory {
442                    return Err(TransactionValidityError::Invalid(
443                        InvalidTransaction::MandatoryValidation,
444                    ));
445                }
446
447                // charge signer for extrinsic storage
448                if let Some(signer) = maybe_signer {
449                    let storage_fees = ExecutiveConfig::LengthToFee::weight_to_fee(
450                        &Weight::from_parts(encoded.len() as u64, 0),
451                    );
452
453                    // best effort to charge the fees to signer.
454                    // if signer does not have enough balance, we continue
455                    let maybe_charged_fees = ExecutiveConfig::Currency::burn_from(
456                        &signer,
457                        storage_fees,
458                        Preservation::Expendable,
459                        Precision::BestEffort,
460                        Fortitude::Force,
461                    );
462
463                    if let Ok(charged_fees) = maybe_charged_fees {
464                        ExecutiveConfig::ExtrinsicStorageFees::on_storage_fees_charged(
465                            charged_fees,
466                            encoded.len() as u32,
467                        )?;
468                    }
469                }
470
471                // note the extrinsic into system storage
472                <frame_system::Pallet<ExecutiveConfig>>::note_extrinsic(encoded);
473
474                // note extrinsic applied. it pays no fees since there was no execution.
475                // we also set the weight to zero since there was no execution done.
476                let r = Err(DispatchErrorWithPostInfo {
477                    post_info: PostDispatchInfo {
478                        actual_weight: None,
479                        pays_fee: Pays::No,
480                    },
481                    error: err,
482                });
483
484                <frame_system::Pallet<ExecutiveConfig>>::note_applied_extrinsic(&r, dispatch_info);
485                Ok(Err(err))
486            }
487        }
488    }

490    // TODO: https://github.com/paritytech/substrate/issues/10711
491    fn final_checks(header: &HeaderFor<ExecutiveConfig>) {
492        sp_tracing::enter_span!(sp_tracing::Level::TRACE, "final_checks");
493        // remove temporaries
494        let new_header = <frame_system::Pallet<ExecutiveConfig>>::finalize();
495
496        // check digest
497        assert_eq!(
498            header.digest().logs().len(),
499            new_header.digest().logs().len(),
500            "Number of digest items must match that calculated."
501        );
502        let items_zip = header
503            .digest()
504            .logs()
505            .iter()
506            .zip(new_header.digest().logs().iter());
507        for (header_item, computed_item) in items_zip {
508            header_item.check_equal(computed_item);
509            assert_eq!(
510                header_item, computed_item,
511                "Digest item must match that calculated."
512            );
513        }
514
515        // check storage root.
516        let storage_root = new_header.state_root();
517        header.state_root().check_equal(storage_root);
518        assert!(
519            header.state_root() == storage_root,
520            "Storage root must match that calculated."
521        );
522
523        assert!(
524            header.extrinsics_root() == new_header.extrinsics_root(),
525            "Transaction trie root must be valid.",
526        );
527    }
