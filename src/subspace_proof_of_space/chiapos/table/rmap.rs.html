1//! Virtual-to-physical mapping table implementation.
2
3#[cfg(test)]
4mod tests;
5
6use crate::chiapos::constants::PARAM_BC;
7use crate::chiapos::table::REDUCED_BUCKET_SIZE;
8use crate::chiapos::table::types::{Position, R};
9
10pub(super) struct Rmap {
11    /// `0` is a sentinel value indicating no virtual pointer is stored yet.
12    ///
13    /// Physical pointer must be increased by `1` to get a virtual pointer before storing. Virtual
14    /// pointer must be decreased by `1` before reading to get a physical pointer.
15    virtual_pointers: [u16; PARAM_BC as usize],
16    positions: [[Position; 2]; REDUCED_BUCKET_SIZE],
17    next_physical_pointer: u16,
18}
19
20impl Rmap {
21    #[inline(always)]
22    pub(super) fn new() -> Self {
23        Self {
24            virtual_pointers: [0; _],
25            positions: [[Position::ZERO; 2]; _],
26            next_physical_pointer: 0,
27        }
28    }
29
30    /// # Safety
31    /// `r` must be in the range `0..PARAM_BC`, there must be at most [`REDUCED_BUCKET_SIZE`] items
32    /// inserted
33    #[inline(always)]
34    unsafe fn insertion_item(&mut self, r: R) -> &mut [Position; 2] {
35        // SAFETY: Guaranteed by function contract
36        let virtual_pointer = unsafe { self.virtual_pointers.get_unchecked_mut(usize::from(r)) };
37
38        if let Some(physical_pointer) = virtual_pointer.checked_sub(1) {
39            // SAFETY: Internal pointers are always valid
40            return unsafe { self.positions.get_unchecked_mut(physical_pointer as usize) };
41        }
42
43        let physical_pointer = self.next_physical_pointer;
44        self.next_physical_pointer += 1;
45        *virtual_pointer = physical_pointer + 1;
46
47        // SAFETY: It is guaranteed by the function contract that the number of added elements will
48        // never exceed `REDUCED_BUCKETS_SIZE`, hence allocated pointers will always be within
49        // bounds
50        unsafe { self.positions.get_unchecked_mut(physical_pointer as usize) }
51    }
52
53    /// Note that `position == Position::ZERO` is effectively ignored here, supporting it cost too
54    /// much in terms of performance and not required for correctness.
55    ///
56    /// # Safety
57    /// `r` must be in the range `0..PARAM_BC`, there must be at most [`REDUCED_BUCKET_SIZE`] items
58    /// inserted
59    #[inline(always)]
60    pub(super) unsafe fn add(&mut self, r: R, position: Position) {
61        // SAFETY: Guaranteed by function contract
62        let rmap_item = unsafe { self.insertion_item(r) };
63
64        // The same `r` can appear in the table multiple times, one duplicate is supported here
65        if rmap_item[0] == Position::ZERO {
66            rmap_item[0] = position;
67        } else if rmap_item[1] == Position::ZERO {
68            rmap_item[1] = position;
69        }
70    }
71
72    /// # Safety
73    /// `r` must be in the range `0..PARAM_BC`
74    #[inline(always)]
75    pub(super) unsafe fn get(&self, r: R) -> [Position; 2] {
76        // SAFETY: Guaranteed by function contract
77        let virtual_pointer = *unsafe { self.virtual_pointers.get_unchecked(usize::from(r)) };
78
79        if let Some(physical_pointer) = virtual_pointer.checked_sub(1) {
80            // SAFETY: Internal pointers are always valid
81            *unsafe { self.positions.get_unchecked(physical_pointer as usize) }
82        } else {
83            [Position::ZERO; 2]
84        }
85    }
86}
