1//! Virtual-to-physical mapping table implementation.
2
3#[cfg(test)]
4mod tests;
5
6use crate::chiapos::constants::PARAM_BC;
7use crate::chiapos::table::REDUCED_BUCKET_SIZE;
8use crate::chiapos::table::types::{Position, R};
9
10pub(super) struct Rmap {
11    /// `0` is a sentinel value indicating no virtual pointer is stored yet.
12    ///
13    /// Physical pointer must be increased by `1` to get a virtual pointer before storing. Virtual
14    /// pointer must be decreased by `1` before reading to get a physical pointer.
15    virtual_pointers: [u16; PARAM_BC as usize],
16    /// `(start_index_in_positions, count)` per distinct r-value.
17    entries: [(u16, u8); REDUCED_BUCKET_SIZE],
18    /// Flat storage for all positions. Positions for the same r-value are consecutive here because
19    /// `add()` is called in Y-sorted bucket iteration order. Within a single bucket, same `r`
20    /// means same `Y` (since `r = y - base` and Y values span exactly one `PARAM_BC` interval),
21    /// so Y-sorted iteration ensures same-`r` additions are consecutive.
22    positions: [Position; REDUCED_BUCKET_SIZE],
23    next_entry: u16,
24    next_position: u16,
25}
26
27impl Rmap {
28    #[inline(always)]
29    pub(super) fn new() -> Self {
30        Self {
31            virtual_pointers: [0; _],
32            entries: [(0, 0); _],
33            positions: [Position::SENTINEL; _],
34            next_entry: 0,
35            next_position: 0,
36        }
37    }
38
39    /// # Safety
40    /// - `r` must be in the range `0..PARAM_BC`
41    /// - There must be at most [`REDUCED_BUCKET_SIZE`] items inserted
42    /// - Additions for the same `r` value must be consecutive (no interleaving with different `r`
43    ///   values between them). This is naturally satisfied when iterating a Y-sorted bucket since
44    ///   same `r` implies same `Y` within a bucket.
45    #[inline(always)]
46    pub(super) unsafe fn add(&mut self, r: R, position: Position) {
47        // SAFETY: Guaranteed by function contract
48        let virtual_pointer = unsafe { self.virtual_pointers.get_unchecked_mut(usize::from(r)) };
49
50        if let Some(physical_pointer) = virtual_pointer.checked_sub(1) {
51            // Existing r-value: increment count, append position
52            // SAFETY: Internal pointers are always valid
53            let entry = unsafe { self.entries.get_unchecked_mut(physical_pointer as usize) };
54            debug_assert!(
55                entry.1 < u8::MAX,
56                "Rmap entry count overflow for r={}",
57                u16::from(r)
58            );
59            entry.1 += 1;
60        } else {
61            // New r-value: allocate entry
62            let physical_pointer = self.next_entry;
63            self.next_entry += 1;
64            *virtual_pointer = physical_pointer + 1;
65
66            // SAFETY: It is guaranteed by the function contract that the number of distinct
67            // r-values will never exceed `REDUCED_BUCKET_SIZE`
68            let entry = unsafe { self.entries.get_unchecked_mut(physical_pointer as usize) };
69            *entry = (self.next_position, 1);
70        }
71
72        // Store position in flat array
73        // SAFETY: Total positions never exceed REDUCED_BUCKET_SIZE
74        unsafe {
75            *self
76                .positions
77                .get_unchecked_mut(self.next_position as usize) = position;
78        }
79        self.next_position += 1;
80    }
81
82    /// Returns all positions for the given r-value as a slice.
83    /// Returns an empty slice if no entry exists.
84    ///
85    /// # Safety
86    /// `r` must be in the range `0..PARAM_BC`
87    #[inline(always)]
88    pub(super) unsafe fn get(&self, r: R) -> &[Position] {
89        // SAFETY: Guaranteed by function contract
90        let virtual_pointer = *unsafe { self.virtual_pointers.get_unchecked(usize::from(r)) };
91
92        if let Some(physical_pointer) = virtual_pointer.checked_sub(1) {
93            // SAFETY: Internal pointers are always valid
94            let &(start_index, count) =
95                unsafe { self.entries.get_unchecked(physical_pointer as usize) };
96            // SAFETY: start_index..start_index+count is always within bounds
97            unsafe {
98                self.positions
99                    .get_unchecked(start_index as usize..start_index as usize + count as usize)
100            }
101        } else {
102            &[]
103        }
104    }
105}
