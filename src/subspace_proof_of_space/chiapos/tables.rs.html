25/// Pick position in `table_number` based on challenge bits
26#[cfg(all(feature = "alloc", any(feature = "full-chiapos", test)))]
27const fn pick_position(
28    [left_position, right_position]: [Position; 2],
29    last_5_challenge_bits: u8,
30    table_number: u8,
31) -> Position {
32    if ((last_5_challenge_bits >> (table_number - 2)) & 1) == 0 {
33        left_position
34    } else {
35        right_position
36    }
37}

77    /// Create Chia proof of space tables. There also exists [`Self::create_parallel()`] that trades
78    /// CPU efficiency and memory usage for lower latency.
79    #[cfg(feature = "alloc")]
80    pub(super) fn create(seed: Seed, cache: &TablesCache) -> Self {
81        let table_1 = Table::<K, 1>::create(seed);
82        let (table_2, _) = Table::<K, 2>::create(table_1, cache);
83        let (table_3, table_2) = Table::<K, 3>::create(table_2, cache);
84        let (table_4, table_3) = Table::<K, 4>::create(table_3, cache);
85        let (table_5, table_4) = Table::<K, 5>::create(table_4, cache);
86        let (table_6, table_5) = Table::<K, 6>::create(table_5, cache);
87        let (table_7, table_6) = Table::<K, 7>::create(table_6, cache);
88
89        Self {
90            table_2,
91            table_3,
92            table_4,
93            table_5,
94            table_6,
95            table_7,
96        }
97    }

122    /// Find proof of space quality for a given challenge
123    #[cfg(all(feature = "alloc", any(feature = "full-chiapos", test)))]
124    pub(super) fn find_quality<'a>(
125        &'a self,
126        challenge: &'a Challenge,
127    ) -> impl Iterator<Item = Quality> + 'a {
128        let last_5_challenge_bits = challenge[challenge.len() - 1] & 0b00011111;
129
130        let first_k_challenge_bits = u32::from_be_bytes(
131            challenge[..size_of::<u32>()]
132                .try_into()
133                .expect("Challenge is known to statically have enough bytes; qed"),
134        ) >> (u32::BITS as usize - usize::from(K));
135
136        // Iterate just over elements that are matching `first_k_challenge_bits` prefix
137        self.table_7.buckets()[Y::bucket_range_from_first_k_bits(first_k_challenge_bits)]
138            .iter()
139            .flat_map(move |positions| {
140                positions
141                    .iter()
142                    .take_while(|&&(position, _y)| position != Position::SENTINEL)
143                    .filter(move |&&(_position, y)| y.first_k_bits() == first_k_challenge_bits)
144            })
145            .map(move |&(position, _y)| {
146                // SAFETY: Internally generated positions that come from the parent table
147                let positions = unsafe { self.table_7.position(position) };
148                // SAFETY: Internally generated positions that come from the parent table
149                let positions = unsafe {
150                    self.table_6
151                        .position(pick_position(positions, last_5_challenge_bits, 6))
152                };
153                // SAFETY: Internally generated positions that come from the parent table
154                let positions = unsafe {
155                    self.table_5
156                        .position(pick_position(positions, last_5_challenge_bits, 5))
157                };
158                // SAFETY: Internally generated positions that come from the parent table
159                let positions = unsafe {
160                    self.table_4
161                        .position(pick_position(positions, last_5_challenge_bits, 4))
162                };
163                // SAFETY: Internally generated positions that come from the parent table
164                let positions = unsafe {
165                    self.table_3
166                        .position(pick_position(positions, last_5_challenge_bits, 3))
167                };
168                // SAFETY: Internally generated positions that come from the parent table
169                let [left_position, right_position] = unsafe {
170                    self.table_2
171                        .position(pick_position(positions, last_5_challenge_bits, 2))
172                };
173
174                // X matches position
175                let left_x = X::from(u32::from(left_position));
176                let right_x = X::from(u32::from(right_position));
177
178                let mut hasher = Sha256::new();
179                hasher.update(challenge);
180                let left_right_xs = (u64::from(left_x) << (u64::BITS as usize - usize::from(K)))
181                    | (u64::from(right_x) << (u64::BITS as usize - usize::from(K * 2)));
182                hasher.update(
183                    &left_right_xs.to_be_bytes()[..(K as usize * 2).div_ceil(u8::BITS as usize)],
184                );
185                hasher.finalize().into()
186            })
187    }

189    /// Find proof of space for a given challenge
190    #[cfg(feature = "alloc")]
191    pub(super) fn find_proof<'a>(
192        &'a self,
193        first_challenge_bytes: [u8; 4],
194    ) -> impl Iterator<Item = [u8; 64 * K as usize / 8]> + 'a {
195        let first_k_challenge_bits =
196            u32::from_be_bytes(first_challenge_bytes) >> (u32::BITS as usize - usize::from(K));
197
198        // Iterate just over elements that are matching `first_k_challenge_bits` prefix
199        self.table_7.buckets()[Y::bucket_range_from_first_k_bits(first_k_challenge_bits)]
200            .iter()
201            .flat_map(move |positions| {
202                positions
203                    .iter()
204                    .take_while(|&&(position, _y)| position != Position::SENTINEL)
205                    .filter(move |&&(_position, y)| y.first_k_bits() == first_k_challenge_bits)
206            })
207            .map(move |&(position, _y)| {
208                // SAFETY: Internally generated positions that come from the parent table
209                let table_6_proof_targets = unsafe { self.table_7.position(position) };
210
211                Self::find_proof_internal(
212                    &self.table_2,
213                    &self.table_3,
214                    &self.table_4,
215                    &self.table_5,
216                    &self.table_6,
217                    table_6_proof_targets,
218                )
219            })
220    }

222    #[cfg(feature = "alloc")]
223    #[inline(always)]
224    fn find_proof_internal(
225        table_2: &PrunedTable<K, 2>,
226        table_3: &PrunedTable<K, 3>,
227        table_4: &PrunedTable<K, 4>,
228        table_5: &PrunedTable<K, 5>,
229        table_6: &PrunedTable<K, 6>,
230        table_6_proof_targets: [Position; 2],
231    ) -> [u8; 64 * K as usize / 8] {
232        let mut proof = [0u8; 64 * K as usize / 8];
233
234        table_6_proof_targets
235            .into_iter()
236            .flat_map(|position| {
237                // SAFETY: Internally generated positions that come from the parent table
238                unsafe { table_6.position(position) }
239            })
240            .flat_map(|position| {
241                // SAFETY: Internally generated positions that come from the parent table
242                unsafe { table_5.position(position) }
243            })
244            .flat_map(|position| {
245                // SAFETY: Internally generated positions that come from the parent table
246                unsafe { table_4.position(position) }
247            })
248            .flat_map(|position| {
249                // SAFETY: Internally generated positions that come from the parent table
250                unsafe { table_3.position(position) }
251            })
252            .flat_map(|position| {
253                // SAFETY: Internally generated positions that come from the parent table
254                unsafe { table_2.position(position) }
255            })
256            .map(|position| {
257                // X matches position
258                X::from(u32::from(position))
259            })
260            .enumerate()
261            .for_each(|(offset, x)| {
262                let x_offset_in_bits = usize::from(K) * offset;
263                // Collect bytes where bits of `x` will be written
264                let proof_bytes = &mut proof[x_offset_in_bits / u8::BITS as usize..]
265                    [..(x_offset_in_bits % u8::BITS as usize + usize::from(K))
266                        .div_ceil(u8::BITS as usize)];
267
268                // Bits of `x` already shifted to the correct location as they will appear
269                // in `proof`
270                let x_shifted = u32::from(x)
271                    << (u32::BITS as usize
272                        - (usize::from(K) + x_offset_in_bits % u8::BITS as usize));
273
274                // Copy `x` bits into proof
275                x_shifted
276                    .to_be_bytes()
277                    .iter()
278                    .zip(proof_bytes)
279                    .for_each(|(from, to)| {
280                        *to |= from;
281                    });
282            });
283
284        proof
285    }

295    /// Verify proof of space for a given seed and challenge
296    pub(super) fn verify_only(
297        seed: &Seed,
298        first_challenge_bytes: [u8; 4],
299        proof_of_space: &[u8; 64 * K as usize / 8],
300    ) -> bool
301    where
302        EvaluatableUsize<{ (K as usize * 2).div_ceil(u8::BITS as usize) }>: Sized,
303    {
304        let first_k_challenge_bits =
305            u32::from_be_bytes(first_challenge_bytes) >> (u32::BITS as usize - usize::from(K));
306
307        let ys_and_metadata = array::from_fn::<_, 64, _>(|offset| {
308            let mut pre_x_bytes = 0u64.to_be_bytes();
309            let offset_in_bits = usize::from(K) * offset;
310            let bytes_to_copy =
311                (offset_in_bits % u8::BITS as usize + usize::from(K)).div_ceil(u8::BITS as usize);
312            // Copy full bytes that contain bits of `x`
313            pre_x_bytes[..bytes_to_copy].copy_from_slice(
314                &proof_of_space[offset_in_bits / u8::BITS as usize..][..bytes_to_copy],
315            );
316            // Extract `pre_x` whose last `K` bits start with `x`
317            let pre_x = u64::from_be_bytes(pre_x_bytes)
318                >> (u64::BITS as usize - (usize::from(K) + offset_in_bits % u8::BITS as usize));
319            // Convert to the desired type and clear extra bits
320            let x = X::from(pre_x as u32 & (u32::MAX >> (u32::BITS as usize - usize::from(K))));
321
322            let y = compute_f1::<K>(x, seed);
323
324            (y, Metadata::from(x))
325        });
326
327        let mut next_ys_and_metadata = [MaybeUninit::uninit(); _];
328        let ys_and_metadata =
329            Self::collect_ys_and_metadata::<2, 1, 64>(&ys_and_metadata, &mut next_ys_and_metadata);
330        let mut next_ys_and_metadata = [MaybeUninit::uninit(); _];
331        let ys_and_metadata =
332            Self::collect_ys_and_metadata::<3, 2, 32>(ys_and_metadata, &mut next_ys_and_metadata);
333        let mut next_ys_and_metadata = [MaybeUninit::uninit(); _];
334        let ys_and_metadata =
335            Self::collect_ys_and_metadata::<4, 3, 16>(ys_and_metadata, &mut next_ys_and_metadata);
336        let mut next_ys_and_metadata = [MaybeUninit::uninit(); _];
337        let ys_and_metadata =
338            Self::collect_ys_and_metadata::<5, 4, 8>(ys_and_metadata, &mut next_ys_and_metadata);
339        let mut next_ys_and_metadata = [MaybeUninit::uninit(); _];
340        let ys_and_metadata =
341            Self::collect_ys_and_metadata::<6, 5, 4>(ys_and_metadata, &mut next_ys_and_metadata);
342        let mut next_ys_and_metadata = [MaybeUninit::uninit(); _];
343        let ys_and_metadata =
344            Self::collect_ys_and_metadata::<7, 6, 2>(ys_and_metadata, &mut next_ys_and_metadata);
345
346        let Some((y, _metadata)) = ys_and_metadata.first() else {
347            return false;
348        };
349
350        // Check if the first K bits of `y` match
351        y.first_k_bits() == first_k_challenge_bits
352    }

354    /// Verify proof of space for a given seed and challenge.
355    ///
356    /// Similar to [`Self::verify_only()`], but also returns quality on successful verification.
357    #[cfg(any(feature = "full-chiapos", test))]
358    pub(super) fn verify(
359        seed: &Seed,
360        challenge: &Challenge,
361        proof_of_space: &[u8; 64 * K as usize / 8],
362    ) -> Option<Quality>
363    where
364        EvaluatableUsize<{ (K as usize * 2).div_ceil(u8::BITS as usize) }>: Sized,
365    {
366        if !Self::verify_only(
367            seed,
368            [challenge[0], challenge[1], challenge[2], challenge[3]],
369            proof_of_space,
370        ) {
371            return None;
372        }
373
374        let last_5_challenge_bits = challenge[challenge.len() - 1] & 0b00011111;
375
376        let mut quality_index = 0_usize.to_be_bytes();
377        quality_index[0] = last_5_challenge_bits;
378        let quality_index = usize::from_be_bytes(quality_index);
379
380        // NOTE: this works correctly but may overflow if `quality_index` is changed to
381        // not be zero-initialized anymore
382        let left_right_xs_bit_offset = quality_index * usize::from(K * 2);
383        // Collect `left_x` and `right_x` bits, potentially with extra bits at the beginning
384        // and the end
385        let left_right_xs_bytes = &proof_of_space[left_right_xs_bit_offset / u8::BITS as usize..]
386            [..(left_right_xs_bit_offset % u8::BITS as usize + usize::from(K * 2))
387                .div_ceil(u8::BITS as usize)];
388
389        let mut left_right_xs = 0u64.to_be_bytes();
390        left_right_xs[..left_right_xs_bytes.len()].copy_from_slice(left_right_xs_bytes);
391        // Move `left_x` and `right_x` bits to most significant bits
392        let left_right_xs =
393            u64::from_be_bytes(left_right_xs) << (left_right_xs_bit_offset % u8::BITS as usize);
394        // Clear extra bits
395        let left_right_xs_mask = u64::MAX << (u64::BITS as usize - usize::from(K * 2));
396        let left_right_xs = left_right_xs & left_right_xs_mask;
397
398        let mut hasher = Sha256::new();
399        hasher.update(challenge);
400        hasher
401            .update(&left_right_xs.to_be_bytes()[..usize::from(K * 2).div_ceil(u8::BITS as usize)]);
402        Some(hasher.finalize().into())
403    }

405    fn collect_ys_and_metadata<
406        'a,
407        const TABLE_NUMBER: u8,
408        const PARENT_TABLE_NUMBER: u8,
409        const N: usize,
410    >(
411        ys_and_metadata: &[(Y, Metadata<K, PARENT_TABLE_NUMBER>)],
412        next_ys_and_metadata: &'a mut [MaybeUninit<(Y, Metadata<K, TABLE_NUMBER>)>; N],
413    ) -> &'a [(Y, Metadata<K, TABLE_NUMBER>)]
414    where
415        EvaluatableUsize<{ metadata_size_bytes(K, TABLE_NUMBER) }>: Sized,
416        EvaluatableUsize<{ metadata_size_bytes(K, PARENT_TABLE_NUMBER) }>: Sized,
417    {
418        let mut next_offset = 0_usize;
419        for &[(left_y, left_metadata), (right_y, right_metadata)] in
420            ys_and_metadata.as_chunks::<2>().0
421        {
422            if !has_match(left_y, right_y) {
423                // Any non-matching pair invalidates the entire proof
424                return &[];
425            }
426
427            next_ys_and_metadata[next_offset].write(compute_fn::<
428                K,
429                TABLE_NUMBER,
430                PARENT_TABLE_NUMBER,
431            >(
432                left_y, left_metadata, right_metadata
433            ));
434            next_offset += 1;
435        }
436
437        // SAFETY: Initialized `next_offset` elements
438        unsafe { next_ys_and_metadata[..next_offset].assume_init_ref() }
439    }
