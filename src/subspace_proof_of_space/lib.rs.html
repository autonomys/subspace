33/// Proof of space table type
34#[derive(Debug, Clone, Copy)]
35pub enum PosTableType {
36    /// Chia table
37    Chia,
38    /// Shim table
39    Shim,
40}

42/// Stateful table generator with better performance.
43///
44/// Prefer cloning it over creating multiple separate generators.
45#[cfg(feature = "alloc")]
46pub trait TableGenerator<T: Table>:
47    fmt::Debug + Default + Clone + Send + Sync + Sized + 'static
48{
49    /// Generate a new table with 32 bytes seed.
50    ///
51    /// There is also [`Self::generate_parallel()`] that can achieve lower latency.
52    fn generate(&self, seed: &PosSeed) -> T;
53
54    /// Generate a new table with 32 bytes seed using parallelism.
55    ///
56    /// This implementation will trade efficiency of CPU and memory usage for lower latency, prefer
57    /// [`Self::generate()`] unless lower latency is critical.
58    #[cfg(feature = "parallel")]
59    fn generate_parallel(&self, seed: &PosSeed) -> T {
60        self.generate(seed)
61    }
62}

64/// Proof of space kind
65pub trait Table: SolutionPotVerifier + Sized + Send + Sync + 'static {
66    /// Proof of space table type
67    const TABLE_TYPE: PosTableType;
68    /// Instance that can be used to generate tables with better performance
69    #[cfg(feature = "alloc")]
70    type Generator: TableGenerator<Self>;
71
72    /// Try to find proof at `challenge_index` if it exists
73    #[cfg(feature = "alloc")]
74    fn find_proof(&self, challenge_index: u32) -> Option<PosProof>;
75
76    /// Check whether proof created earlier is valid
77    fn is_proof_valid(seed: &PosSeed, challenge_index: u32, proof: &PosProof) -> bool;
78
79    /// Returns a stateful table generator with better performance
80    #[cfg(feature = "alloc")]
81    fn generator() -> Self::Generator {
82        Self::Generator::default()
83    }
