1//! Chernoff lower-tail threshold for Binomial(S, p).
2//!
3//! Threshold (additive Chernoff, lower tail):
4//!   Let μ = S * p, where S is slots_in_epoch and p is per-slot win probability.
5//!   t = sqrt( 2 * μ * ln(1/τ) )
6//!   r = floor( μ - t ), clamped to [0, S]
7//!
8//! Property:
9//!   For X ~ Binomial(S, p), `P[ X < r ] <= τ`  (conservative bound)
10//!   This holds for all p in `[0,1]` and any S.
11//!
12//! Notes:
13//! - Provide ln(1/τ) as a FixedU128 constant (precomputed off-chain).
14//! - Examples:
15//!   τ = 1%   => ln(100)   ≈ 4.605170185988092  -> inner ≈ 4_605_170_185_988_091_904
16//!   τ = 0.5% => ln(200)   ≈ 5.298317366548036  -> inner ≈ 5_298_317_366_548_036_000
17//!   τ = 0.1% => ln(1000)  ≈ 6.907755278982137  -> inner ≈ 6_907_755_278_982_137_000
18//! - To decide throughput relevance before calling this (cheap filter), compute
19//!   μ_floor = floor(S * p) and require μ_floor >= E_relevance (e.g., >= 10 for τ=1%).

41/// Threshold for number of bundles operator can produce based on their stake share.
42/// Operators whose expected bundles from their threshold < E_BASE will not be checked
43/// since they are not throughput relevant.
44pub const E_BASE: u64 = 3;

46/// Extension trait for FixedU128 providing rounding to u64.
47trait FixedU128Ext {
48    /// Floor FixedU128 to u64.
49    fn to_u64_floor(self) -> u64;
50    /// Ceil FixedU128 to u64.
51    fn to_u64_ceil(self) -> u64;
52}
53
54impl FixedU128Ext for FixedU128 {
55    fn to_u64_floor(self) -> u64 {
56        (self.into_inner() / FixedU128::DIV).saturated_into()
57    }
58
59    fn to_u64_ceil(self) -> u64 {
60        self.into_inner().div_ceil(FixedU128::DIV).saturated_into()
61    }
62}

64// Exact per-slot win probability is p = threshold / 2^128.
65// Convert that to FixedU128 (1e18 scaling) using U256 intermediates.
66fn p_from_threshold(threshold: u128) -> FixedU128 {
67    // inner = floor(threshold * 1e18 / 2^128)
68    let num = U256::from(threshold).saturating_mul(U256::from(FixedU128::DIV));
69    let den = U256::from(1u128) << 128; // 2^128
70    FixedU128::from_inner((num / den).unique_saturated_into())
71}

73/// Chernoff lower-tail threshold r for Binomial(S, p), with false-positive at most τ.
74///
75/// Inputs:
76/// - slots_in_epoch: S (number of slots in the epoch)
77/// - p_slot: per-slot success probability p in FixedU128 (1e18 scaling)
78/// - ln_one_over_tau: ln(1/τ) in FixedU128 (1e18 scaling), precomputed off-chain
79///
80/// Returns:
81/// - r in [0, S], as u64
82///
83/// Formula:
84///   μ = S * p
85///   r = floor( μ - sqrt( 2 * μ * ln(1/τ) ) ), clamped to [0, S]
86///
87/// Guarantees:
88///   P_honest(X < r) <= τ for X ~ Binomial(S, p). Conservative (safe) bound.
89///
90/// Usage:
91///   let r = chernoff_threshold_fp(S, p, ln_1_over_tau);
92///   if observed_bundles < r { exclude } else { keep }
93fn chernoff_threshold_fp(
94    slots_in_epoch: u64,
95    p_slot: FixedU128,
96    ln_one_over_tau: FixedU128,
97) -> Option<u64> {
98    if slots_in_epoch == 0 || p_slot.is_zero() {
99        return Some(0);
100    }
101
102    if p_slot >= FixedU128::one() {
103        return Some(slots_in_epoch);
104    }
105
106    // μ = S * p
107    let mu = p_slot.saturating_mul(FixedU128::saturating_from_integer(slots_in_epoch));
108
109    // t = sqrt( 2 * μ * ln(1/τ) )
110    let t = FixedU128::from(2u128)
111        .saturating_mul(mu)
112        .saturating_mul(ln_one_over_tau)
113        .try_sqrt()?;
114
115    // r = floor( μ - t ), clamped to [0, S]
116    Some(
117        if mu > t {
118            mu.saturating_sub(t).to_u64_floor()
119        } else {
120            0
121        }
122        .clamp(0, slots_in_epoch),
123    )
124}

126/// Check if an operator is "throughput-relevant" this epoch:
127/// μ_floor = floor(S * p) >= E_relevance
128///
129/// Recommendation for τ = 1%:
130///   E_relevance >= ceil( 2 * ln(1/τ) ) = ceil(9.21) = 10
131fn is_throughput_relevant_fp(slots_in_epoch: u64, p_slot: FixedU128, e_relevance: u64) -> bool {
132    let mu = p_slot.saturating_mul(FixedU128::saturating_from_integer(slots_in_epoch));
133    let mu_floor = mu.to_u64_floor();
134    mu_floor >= e_relevance
135}
136
137// Compute E_relevance = ceil( max( E_BASE, 2 * ln(1/τ) ) ) in integer bundles.
138fn compute_e_relevance(ln_one_over_tau: FixedU128, e_base: u64) -> u64 {
139    let two_ln = FixedU128::saturating_from_integer(2)
140        .saturating_mul(ln_one_over_tau)
141        .to_u64_ceil();
142    max(e_base, two_ln)
143}

145/// Expectations for one operator at epoch end (slots-based, Chernoff-calibrated).
146#[derive(
147    TypeInfo,
148    Debug,
149    Encode,
150    Decode,
151    Clone,
152    PartialEq,
153    Eq,
154    Serialize,
155    Deserialize,
156    DecodeWithMemTracking,
157)]
158pub struct OperatorEpochExpectations {
159    /// floor(μ) = floor(S * p_slot_exact): integer expected bundles this epoch.
160    pub expected_bundles: u64,
161    /// Chernoff lower-bound r: minimum bundles to pass with false-positive ≤ τ.
162    pub min_required_bundles: u64,
163}

165/// Compute epoch-end expectations for an operator using the exact VRF threshold.
166/// Returns None if the operator is not throughput-relevant this epoch.
167///
168/// Flow is:
169/// - calculate_threshold(...) -> u128 threshold
170/// - p_from_threshold(threshold) -> FixedU128 per-slot probability
171/// - compute_e_relevance(ln_one_over_tau, e_base) -> u64 E_relevance
172/// - is_throughput_relevant_fp(S, p, E_relevance) -> bool
173/// - chernoff_threshold_fp(S, p, ln_one_over_tau) -> u64 r
174///
175/// Inputs:
176/// - slots_in_epoch: S
177/// - operator_stake, total_domain_stake: epoch-start stake snapshot
178/// - bundle_slot_probability: (theta_num, theta_den)
179/// - ln_one_over_tau: ln(1/τ) in FixedU128 (e.g., LN_1_OVER_TAU_1_PERCENT)
180/// - e_base: soft relevance floor (e.g., E_BASE)
181pub fn operator_expected_bundles_in_epoch(
182    slots_in_epoch: u64,
183    operator_stake: StakeWeight,
184    total_domain_stake: StakeWeight,
185    bundle_slot_probability: (u64, u64), // (theta_num, theta_den)
186    ln_one_over_tau: FixedU128,
187    e_base: u64,
188) -> Option<OperatorEpochExpectations> {
189    if slots_in_epoch.is_zero() {
190        return None;
191    }
192
193    // Exact per-slot probability from runtime VRF threshold
194    let threshold =
195        calculate_threshold(operator_stake, total_domain_stake, bundle_slot_probability)?;
196    let p_slot = p_from_threshold(threshold);
197    if p_slot.is_zero() {
198        return None;
199    }
200
201    // Throughput relevance: μ_floor >= E_relevance
202    let e_rel = compute_e_relevance(ln_one_over_tau, e_base);
203    if !is_throughput_relevant_fp(slots_in_epoch, p_slot, e_rel) {
204        return None;
205    }
206
207    // Expected bundles (floor μ) and Chernoff lower-bound r
208    let mu_fp = p_slot.saturating_mul(FixedU128::saturating_from_integer(slots_in_epoch));
209    let expected_bundles = mu_fp.to_u64_floor();
210    let min_required_bundles = chernoff_threshold_fp(slots_in_epoch, p_slot, ln_one_over_tau)?;
211
212    Some(OperatorEpochExpectations {
213        expected_bundles,
214        min_required_bundles,
215    })
