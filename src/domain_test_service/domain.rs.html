502    /// Take and stop the domain node and delete its database lock file.
503    ///
504    /// Stopping and restarting a node can cause weird race conditions, with errors like:
505    /// "The system cannot find the path specified".
506    /// If this happens, try increasing the wait time in this method.
507    pub async fn stop(self) -> Result<(), std::io::Error> {
508        let lock_file_path = self.base_path.path().join("paritydb").join("lock");
509        // On Windows, sometimes open files cant be deleted so `drop` first then delete
510        std::mem::drop(self);
511
512        // Give the node time to cleanup, exit, and release the lock file.
513        // TODO: fix the underlying issue or wait for the actual shutdown instead
514        sleep(Duration::from_secs(2)).await;
515
516        // The lock file already being deleted is not a fatal test error, so just log it
517        if let Err(err) = std::fs::remove_file(lock_file_path) {
518            tracing::error!("deleting paritydb lock file failed: {err:?}");
519        }
520        Ok(())
521    }

523    /// Remove all tx from the tx pool
524    pub async fn clear_tx_pool(&self) {
525        let tx_hashes: Vec<_> = self
526            .operator
527            .transaction_pool
528            .ready()
529            .map(|t| self.operator.transaction_pool.hash_of(&t.data))
530            .collect();
531
532        let hash_and_number = HashAndNumber {
533            number: self.client.info().best_number,
534            hash: self.client.info().best_hash,
535        };
536        self.operator
537            .transaction_pool
538            .pool()
539            .prune_known(&hash_and_number, &tx_hashes);
540
541        // `ban_time` have set to 0, explicitly wait 1ms here to ensure `clear_stale` will remove
542        // all the bans as the ban time must be passed.
543        tokio::time::sleep(std::time::Duration::from_millis(1)).await;
544        self.operator
545            .transaction_pool
546            .pool()
547            .validated_pool()
548            .clear_stale(&hash_and_number);
549    }
