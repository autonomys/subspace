47/// Finalizes the domain's current epoch and begins the next epoch.
48/// Returns true of the epoch indeed was finished and the number of operator processed.
49pub(crate) fn do_finalize_domain_current_epoch<T: Config>(
50    domain_id: DomainId,
51) -> Result<EpochTransitionResult, Error> {
52    // Reset pending staking operation count to 0
53    PendingStakingOperationCount::<T>::set(domain_id, 0);
54
55    // re stake operator's tax from the rewards
56    let OperatorTakeRewardTaxResult {
57        rewarded_operator_count,
58        operators_with_self_deposits,
59    } = operator_take_reward_tax_and_stake::<T>(domain_id)?;
60
61    // finalize any withdrawals and then deposits
62    let (completed_epoch_index, finalized_operator_count) =
63        do_finalize_domain_epoch_staking::<T>(domain_id, operators_with_self_deposits)?;
64
65    Ok(EpochTransitionResult {
66        rewarded_operator_count,
67        finalized_operator_count,
68        completed_epoch_index,
69    })
70}

78/// Operator takes `NominationTax` of the current epoch rewards and stake them.
79pub(crate) fn operator_take_reward_tax_and_stake<T: Config>(
80    domain_id: DomainId,
81) -> Result<OperatorTakeRewardTaxResult, Error> {
82    let mut rewarded_operator_count = 0;
83    // a list of operators got rewarded this epoch and their nomination_tax > 0
84    // so they take nomination_tax from the rewards and does a self deposit.
85    // this will lead to `deposits_in_epoch` for the operator
86    // which is same as a nominator deposits to this operator.
87    let mut operators_with_self_deposits = BTreeSet::new();
88    DomainStakingSummary::<T>::try_mutate(domain_id, |maybe_domain_stake_summary| {
89        let stake_summary = maybe_domain_stake_summary
90            .as_mut()
91            .ok_or(TransitionError::DomainNotInitialized)?;
92
93        let mut to_treasury = BalanceOf::<T>::zero();
94        let mut maybe_reward_per_operator = None;
95
96        let domain_rewards = DomainChainRewards::<T>::take(domain_id);
97        // active operators are the operators who received rewards in the epoch
98        // since they are the ones who got rewarded the protocol fees through XDM
99        // once the ER that held this reward is confirmed.
100        // This is also inline with how we reward operators once a ER is confirmed and block fees
101        // distributed to operators who are part of the confirmed ER.
102        let active_operator_count = stake_summary.current_epoch_rewards.len() as u64;
103        match (active_operator_count > 0, !domain_rewards.is_zero()) {
104            // active operators exist and rewards are non-zero
105            (true, true) => {
106                let reward_per_operator = Perquintill::from_rational(1, active_operator_count).mul_floor(domain_rewards);
107                let total_allocated_rewards = reward_per_operator.saturating_mul(BalanceOf::<T>::from(active_operator_count));
108                maybe_reward_per_operator = Some(reward_per_operator);
109                to_treasury = domain_rewards.saturating_sub(total_allocated_rewards);
110            }
111
112            // no active operators but non-zero rewards
113            (false, true) => {
114                to_treasury = domain_rewards
115            }
116
117            // other cases are irrelevant here
118            _ => {}
119        }
120
121
122        while let Some((operator_id, mut reward)) = stake_summary.current_epoch_rewards.pop_first() {
123            reward = reward.saturating_add(maybe_reward_per_operator.unwrap_or_default());
124            Operators::<T>::try_mutate(operator_id, |maybe_operator| {
125                let operator = match maybe_operator.as_mut() {
126                    // It is possible that operator may have de registered and unlocked by the time they
127                    // got rewards, in this case, move the reward to the treasury
128                    None => {
129                        to_treasury += reward;
130                        return Ok(());
131                    }
132                    // Move the reward of slashed and pending slash operator to the treasury
133                    // but do not include operators who are marked invalid bundle authors yet
134                    // as they are not proven until er that marked them as invalid is confirmed.
135                    Some(operator) if matches!(*operator.status::<T>(operator_id), OperatorStatus::Slashed | OperatorStatus::PendingSlash) => {
136                        to_treasury += reward;
137                        return Ok(());
138                    }
139                    Some(operator) => operator,
140                };
141
142                if let Some(reward_per_operator) = maybe_reward_per_operator {
143                    Pallet::<T>::deposit_event(Event::OperatorRewarded {
144                        source: OperatorRewardSource::XDMProtocolFees,
145                        operator_id,
146                        reward: reward_per_operator,
147                    });
148                }
149
150                // calculate operator tax, mint the balance, and stake them
151                let operator_tax_amount = operator.nomination_tax.mul_floor(reward);
152                if !operator_tax_amount.is_zero() {
153                    let nominator_id = OperatorIdOwner::<T>::get(operator_id)
154                        .ok_or(TransitionError::MissingOperatorOwner)?;
155                    T::Currency::mint_into(&nominator_id, operator_tax_amount)
156                        .map_err(|_| TransitionError::MintBalance)?;
157
158                    // Reserve for the bundle storage fund
159                    let operator_tax_deposit =
160                        deposit_reserve_for_storage_fund::<T>(operator_id, &nominator_id, operator_tax_amount)
161                            .map_err(TransitionError::BundleStorageFund)?;
162
163                    crate::staking::hold_deposit::<T>(
164                        &nominator_id,
165                        operator_id,
166                        operator_tax_deposit.staking,
167                    )?;
168
169                    // increment total deposit for operator pool within this epoch
170                    operator.deposits_in_epoch = operator
171                        .deposits_in_epoch
172                        .checked_add(&operator_tax_deposit.staking)
173                        .ok_or(TransitionError::BalanceOverflow)?;
174
175                    // Increase total storage fee deposit as there is new deposit to the storage fund
176                    operator.total_storage_fee_deposit = operator
177                        .total_storage_fee_deposit
178                        .checked_add(&operator_tax_deposit.storage_fee_deposit)
179                        .ok_or(TransitionError::BalanceOverflow)?;
180
181                    let current_domain_epoch = (domain_id, stake_summary.current_epoch_index).into();
182                    crate::staking::do_calculate_previous_epoch_deposit_shares_and_add_new_deposit::<T>(
183                        operator_id,
184                        nominator_id,
185                        current_domain_epoch,
186                        operator_tax_deposit,
187                        None,
188                    )?;
189
190                    Pallet::<T>::deposit_event(Event::OperatorTaxCollected {
191                        operator_id,
192                        tax: operator_tax_amount,
193                    });
194
195                    operators_with_self_deposits.insert(operator_id);
196                }
197
198                // Add the remaining rewards to the operator's `current_total_stake` which increases the
199                // share price of the staking pool and as a way to distribute the reward to the nominator
200                let rewards = reward
201                    .checked_sub(&operator_tax_amount)
202                    .ok_or(TransitionError::BalanceUnderflow)?;
203
204                operator.current_total_stake = operator
205                    .current_total_stake
206                    .checked_add(&rewards)
207                    .ok_or(TransitionError::BalanceOverflow)?;
208
209                rewarded_operator_count += 1;
210
211                Ok(())
212            })?;
213        }
214
215        mint_into_treasury::<T>(to_treasury)?;
216
217        Ok(())
218    })
219        .map_err(Error::OperatorRewardStaking)?;
220
221    Ok(OperatorTakeRewardTaxResult {
222        rewarded_operator_count,
223        operators_with_self_deposits,
224    })
225}

227pub(crate) fn do_finalize_domain_epoch_staking<T: Config>(
228    domain_id: DomainId,
229    operators_with_self_deposits: BTreeSet<OperatorId>,
230) -> Result<(EpochIndex, u32), Error> {
231    let mut finalized_operator_count = 0;
232    DomainStakingSummary::<T>::try_mutate(domain_id, |maybe_stake_summary| {
233        let stake_summary = maybe_stake_summary
234            .as_mut()
235            .ok_or(TransitionError::DomainNotInitialized)?;
236
237        let previous_epoch = stake_summary.current_epoch_index;
238        let next_epoch = previous_epoch
239            .checked_add(One::one())
240            .ok_or(TransitionError::EpochOverflow)?;
241
242        let mut total_domain_stake = BalanceOf::<T>::zero();
243        let mut current_operators = BTreeMap::new();
244        let mut next_operators = BTreeSet::new();
245        let current_slot = T::BlockSlot::current_slot();
246        let total_epoch_slots = EpochStartSlot::<T>::get()
247            .map(|start_slot| current_slot - start_slot + 1)
248            .unwrap_or_default();
249        let epoch_total_stake = stake_summary.current_total_stake;
250        let operators_total_bundle_count =
251            OperatorBundleCountInEpoch::<T>::drain().collect::<BTreeMap<_, _>>();
252        let bundle_slot_probability = DomainRegistry::<T>::get(domain_id)
253            .ok_or(TransitionError::DomainNotInitialized)?
254            .domain_config
255            .bundle_slot_probability;
256        for next_operator_id in &stake_summary.next_operators {
257            // If an operator is pending to slash then similar to the slashed operator it should not be added
258            // into the `next_operators/current_operators` and we should not `do_finalize_operator_epoch_staking`
259            // for it.
260            if Pallet::<T>::is_operator_pending_to_slash(domain_id, *next_operator_id) {
261                continue;
262            }
263
264            // check operator performance in the previous epoch if the operator was part of the previous epoch set
265            // if they are not part of the previous epoch set, their performance will be checked in the next epoch.
266            if let Some(operator_stake) = stake_summary.current_operators.get(next_operator_id)
267                && let Some(bundle_count) = operators_total_bundle_count.get(next_operator_id)
268            {
269                let maybe_operator_epoch_expectations = operator_expected_bundles_in_epoch(
270                    total_epoch_slots.into(),
271                    (*operator_stake).saturated_into(),
272                    epoch_total_stake.saturated_into(),
273                    bundle_slot_probability,
274                    LN_1_OVER_TAU_0_5_PERCENT,
275                    E_BASE,
276                );
277
278                if let Some(operator_epoch_expectations) = maybe_operator_epoch_expectations
279                    && bundle_count < &operator_epoch_expectations.min_required_bundles
280                {
281                    Pallet::<T>::deposit_event(Event::OperatorOffline {
282                        operator_id: *next_operator_id,
283                        domain_id,
284                        submitted_bundles: *bundle_count,
285                        expectations: operator_epoch_expectations,
286                    });
287                };
288            }
289
290            let (operator_stake, stake_changed) = do_finalize_operator_epoch_staking::<T>(
291                domain_id,
292                *next_operator_id,
293                previous_epoch,
294                true,
295            )?;
296
297            total_domain_stake = total_domain_stake
298                .checked_add(&operator_stake)
299                .ok_or(TransitionError::BalanceOverflow)?;
300            current_operators.insert(*next_operator_id, operator_stake);
301            next_operators.insert(*next_operator_id);
302
303            if stake_changed {
304                finalized_operator_count += 1;
305            }
306        }
307
308        // set epoch start slot
309        EpochStartSlot::<T>::put(current_slot);
310
311        // we finalized the operators who are in the next operator set.
312        // But there will be deposits/withdrawals for operators who are not part of the next operator set.
313        // So they need to have share price for the previous epoch
314        //  - Deregistered operators who got new deposits/withdrawals before they de-registered.
315        //  - Deactivated operators who got new deposits/withdrawals before they were deactivated.
316        //  - Deactivated operators who got new withdrawals after they were deactivated.
317        //  - InvalidBundle authors can have deposits or withdrawals before they are marked invalid.
318        //  - Any operator who received rewards in the previous epoch
319        let mut operators_to_calculate_share_price = operators_with_self_deposits;
320        // include invalid bundle authors
321        operators_to_calculate_share_price.extend(InvalidBundleAuthors::<T>::take(domain_id));
322        // include operators who de-registered in this epoch
323        operators_to_calculate_share_price.extend(DeregisteredOperators::<T>::take(domain_id));
324        // include operators who we deactivated in this epoch
325        operators_to_calculate_share_price.extend(DeactivatedOperators::<T>::take(domain_id));
326        // exclude operators who have already been processed as they are in next operator set.
327        operators_to_calculate_share_price.retain(|&x| !stake_summary.next_operators.contains(&x));
328
329        for operator_id in operators_to_calculate_share_price {
330            if let Some(operator) = Operators::<T>::get(operator_id) {
331                if matches!(
332                    *operator.status::<T>(operator_id),
333                    OperatorStatus::Slashed | OperatorStatus::PendingSlash
334                ) {
335                    // operator is either slashed.
336                    // no need to calculate share price for them
337                    continue;
338                }
339
340                let (_, stake_changed) = do_finalize_operator_epoch_staking::<T>(
341                    domain_id,
342                    operator_id,
343                    previous_epoch,
344                    false,
345                )?;
346
347                if stake_changed {
348                    finalized_operator_count += 1;
349                }
350            }
351        }
352
353        let election_verification_params = ElectionVerificationParams {
354            operators: stake_summary.current_operators.clone(),
355            total_domain_stake: stake_summary.current_total_stake,
356        };
357
358        LastEpochStakingDistribution::<T>::insert(domain_id, election_verification_params);
359
360        stake_summary.current_epoch_index = next_epoch;
361        stake_summary.current_total_stake = total_domain_stake;
362        stake_summary.current_operators = current_operators;
363        stake_summary.next_operators = next_operators;
364
365        Ok((previous_epoch, finalized_operator_count))
366    })
367    .map_err(Error::FinalizeDomainEpochStaking)
368}

370/// Finalize the epoch for the operator
371///
372/// Return the new total stake of the operator and a bool indicate if its total stake
373/// is changed due to deposit/withdraw/reward happened in the previous epoch
374pub(crate) fn do_finalize_operator_epoch_staking<T: Config>(
375    domain_id: DomainId,
376    operator_id: OperatorId,
377    previous_epoch: EpochIndex,
378    check_status: bool,
379) -> Result<(BalanceOf<T>, bool), TransitionError> {
380    let mut operator = match Operators::<T>::get(operator_id) {
381        Some(op) => op,
382        None => return Err(TransitionError::UnknownOperator),
383    };
384
385    if check_status && *operator.status::<T>(operator_id) != OperatorStatus::Registered {
386        return Err(TransitionError::OperatorNotRegistered);
387    }
388
389    // if there are no deposits, withdrawals, and epoch rewards for this operator
390    // then short-circuit and return early.
391    if operator.deposits_in_epoch.is_zero() && operator.withdrawals_in_epoch.is_zero() {
392        return Ok((operator.current_total_stake, false));
393    }
394
395    let mut total_stake = operator.current_total_stake;
396    let mut total_shares = operator.current_total_shares;
397    let share_price = SharePrice::new::<T>(total_shares, total_stake)?;
398
399    // calculate and subtract total withdrew shares from previous epoch
400    if !operator.withdrawals_in_epoch.is_zero() {
401        let withdraw_stake = share_price.shares_to_stake::<T>(operator.withdrawals_in_epoch);
402        total_stake = total_stake
403            .checked_sub(&withdraw_stake)
404            .ok_or(TransitionError::BalanceUnderflow)?;
405        total_shares = total_shares
406            .checked_sub(&operator.withdrawals_in_epoch)
407            .ok_or(TransitionError::ShareUnderflow)?;
408
409        operator.withdrawals_in_epoch = Zero::zero();
410    };
411
412    // calculate and add total deposits from the previous epoch
413    if !operator.deposits_in_epoch.is_zero() {
414        let deposited_shares = share_price.stake_to_shares::<T>(operator.deposits_in_epoch);
415        total_stake = total_stake
416            .checked_add(&operator.deposits_in_epoch)
417            .ok_or(TransitionError::BalanceOverflow)?;
418        total_shares = total_shares
419            .checked_add(&deposited_shares)
420            .ok_or(TransitionError::ShareOverflow)?;
421
422        operator.deposits_in_epoch = Zero::zero();
423    };
424
425    // update operator pool epoch share price
426    // TODO: once we have reference counting, we do not need to
427    //  store this for every epoch for every operator but instead
428    //  store only those share prices of operators which has either a deposit or withdraw
429    OperatorEpochSharePrice::<T>::insert(
430        operator_id,
431        DomainEpoch::from((domain_id, previous_epoch)),
432        share_price,
433    );
434
435    // update operator state
436    operator.current_total_shares = total_shares;
437    operator.current_total_stake = total_stake;
438    Operators::<T>::set(operator_id, Some(operator));
439
440    Ok((total_stake, true))
441}

443pub(crate) fn mint_funds<T: Config>(
444    account_id: &T::AccountId,
445    amount_to_mint: BalanceOf<T>,
446) -> Result<(), TransitionError> {
447    if !amount_to_mint.is_zero() {
448        T::Currency::mint_into(account_id, amount_to_mint)
449            .map_err(|_| TransitionError::MintBalance)?;
450    }
451
452    Ok(())
453}

455pub(crate) fn mint_into_treasury<T: Config>(amount: BalanceOf<T>) -> Result<(), TransitionError> {
456    if amount.is_zero() {
457        return Ok(());
458    }
459
460    let total_funds = AccumulatedTreasuryFunds::<T>::get()
461        .checked_add(&amount)
462        .ok_or(TransitionError::BalanceOverflow)?;
463
464    match T::Currency::can_deposit(&T::TreasuryAccount::get(), total_funds, Provenance::Minted) {
465        // Deposit is possible, so we mint the funds into treasury.
466        DepositConsequence::Success => {
467            T::Currency::mint_into(&T::TreasuryAccount::get(), total_funds)
468                .map_err(|_| TransitionError::MintBalance)?;
469            AccumulatedTreasuryFunds::<T>::kill();
470        }
471        // Deposit cannot be done to treasury, so hold the funds until we can.
472        _ => AccumulatedTreasuryFunds::<T>::set(total_funds),
473    }
474    Ok(())
475}

477/// Slashes any pending slashed operators.
478/// At max slashes the `max_nominator_count` under given operator
479pub(crate) fn do_slash_operator<T: Config>(
480    domain_id: DomainId,
481    max_nominator_count: u32,
482) -> Result<u32, TransitionError> {
483    let mut slashed_nominator_count = 0u32;
484    let (operator_id, slashed_operators) = match PendingSlashes::<T>::get(domain_id) {
485        None => return Ok(0),
486        Some(mut slashed_operators) => match slashed_operators.pop_first() {
487            None => {
488                PendingSlashes::<T>::remove(domain_id);
489                return Ok(0);
490            }
491            Some(operator_id) => (operator_id, slashed_operators),
492        },
493    };
494
495    let current_domain_epoch_index = DomainStakingSummary::<T>::get(domain_id)
496        .ok_or(TransitionError::DomainNotInitialized)?
497        .current_epoch_index;
498
499    Operators::<T>::try_mutate_exists(operator_id, |maybe_operator| {
500        // take the operator so this operator info is removed once we slash the operator.
501        let mut operator = maybe_operator
502            .take()
503            .ok_or(TransitionError::UnknownOperator)?;
504
505        let staked_hold_id = T::HoldIdentifier::staking_staked();
506
507        let mut total_stake = operator.current_total_stake;
508        let mut total_shares = operator.current_total_shares;
509        let share_price = SharePrice::new::<T>(total_shares, total_stake)?;
510
511        let mut total_storage_fee_deposit = operator.total_storage_fee_deposit;
512
513        // transfer all the staked funds to the treasury account
514        // any gains will be minted to treasury account
515        for (nominator_id, mut deposit) in Deposits::<T>::drain_prefix(operator_id) {
516            let locked_amount = DepositOnHold::<T>::take((operator_id, nominator_id.clone()));
517
518            // convert any previous epoch deposits
519            match do_convert_previous_epoch_deposits::<T>(
520                operator_id,
521                &mut deposit,
522                current_domain_epoch_index,
523            ) {
524                // Share price may be missing if there is deposit happen in the same epoch as slash
525                Ok(()) | Err(TransitionError::MissingOperatorEpochSharePrice) => {}
526                Err(err) => return Err(err),
527            }
528
529            // there maybe some withdrawals that are initiated in this epoch where operator was slash
530            // then collect and include them to find the final stake amount
531            let (
532                amount_ready_to_withdraw,
533                withdraw_storage_fee_on_hold,
534                shares_withdrew_in_current_epoch,
535            ) = Withdrawals::<T>::take(operator_id, nominator_id.clone())
536                .map(|mut withdrawal| {
537                    match do_convert_previous_epoch_withdrawal::<T>(
538                        operator_id,
539                        &mut withdrawal,
540                        current_domain_epoch_index,
541                    ) {
542                        // Share price may be missing if there is withdrawal happen in the same epoch as slash
543                        Ok(()) | Err(TransitionError::MissingOperatorEpochSharePrice) => {}
544                        Err(err) => return Err(err),
545                    }
546                    Ok((
547                        withdrawal.total_withdrawal_amount,
548                        withdrawal.total_storage_fee_withdrawal,
549                        withdrawal
550                            .withdrawal_in_shares
551                            .map(|WithdrawalInShares { shares, .. }| shares)
552                            .unwrap_or_default(),
553                    ))
554                })
555                .unwrap_or(Ok((Zero::zero(), Zero::zero(), Zero::zero())))?;
556
557            // include all the known shares and shares that were withdrawn in the current epoch
558            let nominator_shares = deposit
559                .known
560                .shares
561                .checked_add(&shares_withdrew_in_current_epoch)
562                .ok_or(TransitionError::ShareOverflow)?;
563            total_shares = total_shares
564                .checked_sub(&nominator_shares)
565                .ok_or(TransitionError::ShareOverflow)?;
566
567            // current staked amount
568            let nominator_staked_amount = share_price.shares_to_stake::<T>(nominator_shares);
569            total_stake = total_stake
570                .checked_sub(&nominator_staked_amount)
571                .ok_or(TransitionError::BalanceOverflow)?;
572
573            let pending_deposit = deposit
574                .pending
575                .map(|pending_deposit| pending_deposit.amount)
576                .unwrap_or_default();
577
578            // do not slash the deposit that is not staked yet
579            let amount_to_slash_in_holding = locked_amount
580                .checked_sub(&pending_deposit)
581                .ok_or(TransitionError::BalanceUnderflow)?;
582
583            T::Currency::transfer_on_hold(
584                &staked_hold_id,
585                &nominator_id,
586                &T::TreasuryAccount::get(),
587                amount_to_slash_in_holding,
588                Precision::Exact,
589                Restriction::Free,
590                Fortitude::Force,
591            )
592            .map_err(|_| TransitionError::RemoveLock)?;
593
594            // release rest of the deposited un staked amount back to nominator
595            T::Currency::release(
596                &staked_hold_id,
597                &nominator_id,
598                pending_deposit,
599                Precision::BestEffort,
600            )
601            .map_err(|_| TransitionError::RemoveLock)?;
602
603            // these are nominator rewards that will be minted to treasury
604            // include amount ready to be withdrawn to calculate the final reward
605            let nominator_reward = nominator_staked_amount
606                .checked_add(&amount_ready_to_withdraw)
607                .ok_or(TransitionError::BalanceOverflow)?
608                .saturating_sub(amount_to_slash_in_holding);
609
610            mint_into_treasury::<T>(nominator_reward)?;
611
612            // Transfer the deposited non-staked storage fee back to nominator
613            if let Some(pending_deposit) = deposit.pending {
614                let storage_fund_redeem_price = bundle_storage_fund::storage_fund_redeem_price::<T>(
615                    operator_id,
616                    total_storage_fee_deposit,
617                );
618
619                bundle_storage_fund::withdraw_to::<T>(
620                    operator_id,
621                    &nominator_id,
622                    storage_fund_redeem_price.redeem(pending_deposit.storage_fee_deposit),
623                )
624                .map_err(TransitionError::BundleStorageFund)?;
625
626                total_storage_fee_deposit =
627                    total_storage_fee_deposit.saturating_sub(pending_deposit.storage_fee_deposit);
628            }
629
630            // Transfer all the storage fee on withdraw to the treasury
631            T::Currency::transfer_on_hold(
632                &T::HoldIdentifier::storage_fund_withdrawal(),
633                &nominator_id,
634                &T::TreasuryAccount::get(),
635                withdraw_storage_fee_on_hold,
636                Precision::Exact,
637                Restriction::Free,
638                Fortitude::Force,
639            )
640            .map_err(|_| TransitionError::RemoveLock)?;
641
642            slashed_nominator_count += 1;
643            if slashed_nominator_count >= max_nominator_count {
644                break;
645            }
646        }
647
648        // The operator state is safe to cleanup if there is no entry in `Deposits` and `Withdrawals`
649        // which means all nominator (including the operator owner) have been slashed.
650        let cleanup_operator = !Deposits::<T>::contains_prefix(operator_id)
651            && !Withdrawals::<T>::contains_prefix(operator_id);
652
653        if cleanup_operator {
654            do_cleanup_operator::<T>(operator_id, total_stake)?;
655            if slashed_operators.is_empty() {
656                PendingSlashes::<T>::remove(domain_id);
657            } else {
658                PendingSlashes::<T>::set(domain_id, Some(slashed_operators));
659            }
660        } else {
661            // set update total shares, total stake and total storage fee deposit for operator
662            operator.current_total_shares = total_shares;
663            operator.current_total_stake = total_stake;
664            operator.total_storage_fee_deposit = total_storage_fee_deposit;
665            *maybe_operator = Some(operator);
666        }
667
668        Ok(slashed_nominator_count)
669    })
670}
