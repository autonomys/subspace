44/// Domain runtime specific information to create domain raw genesis.
45#[derive(TypeInfo, Debug, Encode, Decode, Clone, PartialEq, Eq, Copy)]
46#[allow(clippy::upper_case_acronyms)]
47pub enum DomainRuntimeInfoV2 {
48    EVM { chain_id: EVMChainId },
49    AutoId,
50}

52#[derive(TypeInfo, Debug, Encode, Decode, Clone, PartialEq, Eq)]
53pub struct DomainObjectV2<Number, ReceiptHash, AccountId: Ord, Balance> {
54    /// The address of the domain creator, used to validate updating the domain config.
55    pub owner_account_id: AccountId,
56    /// The consensus chain block number when the domain first instantiated.
57    pub created_at: Number,
58    /// The hash of the genesis execution receipt for this domain.
59    pub genesis_receipt_hash: ReceiptHash,
60    /// The domain config.
61    pub domain_config: DomainConfig<AccountId, Balance>,
62    /// Domain runtime specific information.
63    pub domain_runtime_info: DomainRuntimeInfoV2,
64    /// The amount of balance hold on the domain owner account
65    pub domain_instantiation_deposit: Balance,
66}
67
68impl<T, U, V: Ord, W> From<DomainObjectV2<T, U, V, W>> for DomainObjectV3<T, U, V, W> {
69    fn from(domain_object_v2: DomainObjectV2<T, U, V, W>) -> Self {
70        let DomainObjectV2 {
71            owner_account_id,
72            created_at,
73            genesis_receipt_hash,
74            domain_config,
75            domain_runtime_info: domain_runtime_info_v2,
76            domain_instantiation_deposit,
77        } = domain_object_v2;
78
79        DomainObjectV3 {
80            owner_account_id,
81            created_at,
82            genesis_receipt_hash,
83            domain_config,
84            domain_runtime_info: domain_runtime_info_v2.into(),
85            domain_instantiation_deposit,
86        }
87    }
88}

90impl From<DomainRuntimeInfoV2> for DomainRuntimeInfoV3 {
91    fn from(domain_runtime_info_v2: DomainRuntimeInfoV2) -> Self {
92        match domain_runtime_info_v2 {
93            DomainRuntimeInfoV2::EVM { chain_id } => DomainRuntimeInfoV3::Evm {
94                chain_id,
95                // Added in V3
96                domain_runtime_config: EvmDomainRuntimeConfigV3::default(),
97            },
98            DomainRuntimeInfoV2::AutoId => DomainRuntimeInfoV3::AutoId {
99                // Added in V3
100                domain_runtime_config: AutoIdDomainRuntimeConfigV3::default(),
101            },
102        }
103    }
104}

128mod domain_registry_structure_migration {
129    use super::{DomainObjectV2, DomainObjectV3, DomainRegistryV3};
130    use crate::{BalanceOf, BlockNumberFor, Config, ReceiptHashFor};
131    use frame_support::pallet_prelude::Weight;
132    use sp_core::Get;
133
134    pub(super) fn migrate_domain_registry_structure<T: Config>() -> Weight {
135        let mut domain_count = 0;
136
137        DomainRegistryV3::<T>::translate_values::<
138            DomainObjectV2<BlockNumberFor<T>, ReceiptHashFor<T>, T::AccountId, BalanceOf<T>>,
139            _,
140        >(|domain_object_v2| {
141            domain_count += 1;
142
143            Some(DomainObjectV3::from(domain_object_v2))
144        });
145
146        // 1 read and 1 write per domain
147        T::DbWeight::get().reads_writes(domain_count, domain_count)
148    }
149}

151#[cfg(test)]
152mod tests {
153    use super::domain_registry_structure_migration::*;
154    use super::*;
155    use crate::tests::{new_test_ext, Test};
156    use sp_domains::{EvmType as EvmTypeV3, OperatorAllowList};
157
158    #[test]
159    fn test_domain_registry_structure_migration_evm() {
160        let mut ext = new_test_ext();
161        let domain_id: DomainId = 0.into();
162        let chain_id = 8u32.into();
163        let domain_config = DomainConfig {
164            domain_name: "test-evm-migrate".to_string(),
165            runtime_id: 3u32,
166            max_bundle_size: 4,
167            max_bundle_weight: 5.into(),
168            bundle_slot_probability: (6, 7),
169            operator_allow_list: OperatorAllowList::Anyone,
170            initial_balances: vec![],
171        };
172        let domain = DomainObjectV2 {
173            owner_account_id: 1u32.into(),
174            created_at: 2u32.into(),
175            genesis_receipt_hash: Default::default(),
176            domain_config: domain_config.clone(),
177            domain_runtime_info: DomainRuntimeInfoV2::EVM { chain_id },
178            domain_instantiation_deposit: 9u32.into(),
179        };
180        let expected_migration = DomainObjectV3 {
181            owner_account_id: domain.owner_account_id,
182            created_at: domain.created_at,
183            genesis_receipt_hash: domain.genesis_receipt_hash,
184            domain_config: domain_config.clone(),
185            domain_runtime_info: DomainRuntimeInfoV3::Evm {
186                chain_id,
187                domain_runtime_config: EvmDomainRuntimeConfigV3 {
188                    // All current EVM domains are public EVMs
189                    evm_type: EvmTypeV3::Public,
190                },
191            },
192            domain_instantiation_deposit: domain.domain_instantiation_deposit,
193        };
194
195        ext.execute_with(|| {
196            DomainRegistryV2::<Test>::set(domain_id, Some(domain.clone()));
197
198            let fallback_outcome = Pallet::<Test>::domain_registry_fallback(domain_id);
199            assert_eq!(fallback_outcome.unwrap(), expected_migration);
200
201            let weights = migrate_domain_registry_structure::<Test>();
202            assert_eq!(
203                weights,
204                <Test as frame_system::Config>::DbWeight::get().reads_writes(1, 1),
205            );
206            assert_eq!(
207                DomainRegistryV3::<Test>::get(domain_id).unwrap(),
208                expected_migration,
209            );
210        });
211    }
