58impl<Runtime> DomainsExtension<Runtime>
59where
60    Runtime: Config + scale_info::TypeInfo + fmt::Debug + Send + Sync,
61{
62    fn do_validate_submit_bundle(
63        opaque_bundle: &OpaqueBundleOf<Runtime>,
64        source: TransactionSource,
65    ) -> TransactionValidity {
66        let pre_dispatch = TransactionSource::InBlock == source;
67        if pre_dispatch {
68            Domains::<Runtime>::validate_submit_bundle(opaque_bundle, true)
69                .map_err(|_| InvalidTransaction::Call.into())
70                .map(|_| ValidTransaction::default())
71        } else {
72            let domain_id = opaque_bundle.domain_id();
73            let operator_id = opaque_bundle.operator_id();
74            let slot_number = opaque_bundle.slot_number();
75
76            if let Err(e) = Domains::<Runtime>::validate_submit_bundle(opaque_bundle, false) {
77                Domains::<Runtime>::log_bundle_error(&e, domain_id, operator_id);
78                return e.into();
79            }
80
81            ValidTransaction::with_tag_prefix("SubspaceSubmitBundle")
82                // Bundle have a bit higher priority than normal extrinsic but must less than
83                // fraud proof
84                .priority(1)
85                .longevity(
86                    Runtime::ConfirmationDepthK::get()
87                        .try_into()
88                        .unwrap_or_else(|_| {
89                            panic!("Block number always fits in TransactionLongevity; qed")
90                        }),
91                )
92                .and_provides((operator_id, slot_number))
93                .propagate(true)
94                .build()
95        }
96    }
97
98    fn do_validate_fraud_proof(
99        fraud_proof: &FraudProofFor<Runtime>,
100        source: TransactionSource,
101    ) -> TransactionValidity {
102        let pre_dispatch = TransactionSource::InBlock == source;
103        if pre_dispatch {
104            Domains::<Runtime>::validate_fraud_proof(fraud_proof)
105                .map(|_| ())
106                .map_err(|_| InvalidTransaction::Call.into())
107                .map(|_| ValidTransaction::default())
108        } else {
109            let (tag, priority) = match Domains::<Runtime>::validate_fraud_proof(fraud_proof) {
110                Err(e) => {
111                    log::warn!(
112                        target: "runtime::domains",
113                        "Bad fraud proof {fraud_proof:?}, error: {e:?}",
114                    );
115                    return InvalidTransactionCode::FraudProof.into();
116                }
117                Ok(tp) => tp,
118            };
119
120            ValidTransaction::with_tag_prefix("SubspaceSubmitFraudProof")
121                .priority(priority)
122                .and_provides(tag)
123                .longevity(TransactionLongevity::MAX)
124                // We need this extrinsic to be propagated to the farmer nodes.
125                .propagate(true)
126                .build()
127        }
128    }
129
130    fn do_validate_singleton_receipt(
131        singleton_receipt: &SingletonReceiptOf<Runtime>,
132        source: TransactionSource,
133    ) -> TransactionValidity {
134        let pre_dispatch = TransactionSource::InBlock == source;
135        if pre_dispatch {
136            Domains::<Runtime>::validate_singleton_receipt(singleton_receipt, true)
137                .map_err(|_| InvalidTransaction::Call.into())
138                .map(|_| ValidTransaction::default())
139        } else {
140            let domain_id = singleton_receipt.domain_id();
141            let operator_id = singleton_receipt.operator_id();
142            let slot_number = singleton_receipt.slot_number();
143
144            if let Err(e) = Domains::<Runtime>::validate_singleton_receipt(singleton_receipt, false)
145            {
146                Domains::<Runtime>::log_bundle_error(&e, domain_id, operator_id);
147                return e.into();
148            }
149
150            ValidTransaction::with_tag_prefix("SubspaceSubmitReceipt")
151                // Receipt have a bit higher priority than normal extrinsic but must less than
152                // fraud proof
153                .priority(1)
154                .longevity(
155                    Runtime::ConfirmationDepthK::get()
156                        .try_into()
157                        .unwrap_or_else(|_| {
158                            panic!("Block number always fits in TransactionLongevity; qed")
159                        }),
160                )
161                .and_provides((operator_id, slot_number))
162                .propagate(true)
163                .build()
164        }
165    }
166}

168impl<Runtime> TransactionExtension<RuntimeCallFor<Runtime>> for DomainsExtension<Runtime>
169where
170    Runtime: Config + scale_info::TypeInfo + fmt::Debug + Send + Sync,
171    <RuntimeCallFor<Runtime> as Dispatchable>::RuntimeOrigin:
172        AsSystemOriginSigner<<Runtime as frame_system::Config>::AccountId> + From<Origin> + Clone,
173    RuntimeCallFor<Runtime>: MaybeDomainsCall<Runtime>,
174{
175    const IDENTIFIER: &'static str = "DomainsExtension";
176    type Implicit = ();
177    type Val = ();
178    type Pre = ();
179
180    fn weight(&self, call: &Runtime::RuntimeCall) -> Weight {
181        // This extension only apply to the following 3 calls thus only return weight for them.
182        match call.maybe_domains_call() {
183            Some(DomainsCall::submit_bundle { .. }) => {
184                <Runtime as Config>::WeightInfo::validate_submit_bundle()
185            }
186            Some(DomainsCall::submit_fraud_proof { .. }) => {
187                // FIXME: proper weight
188                Weight::zero()
189            }
190            Some(DomainsCall::submit_receipt { .. }) => {
191                <Runtime as Config>::WeightInfo::validate_singleton_receipt()
192            }
193            _ => Weight::zero(),
194        }
195    }
196
197    fn validate(
198        &self,
199        origin: DispatchOriginOf<RuntimeCallFor<Runtime>>,
200        call: &RuntimeCallFor<Runtime>,
201        _info: &DispatchInfoOf<RuntimeCallFor<Runtime>>,
202        _len: usize,
203        _self_implicit: Self::Implicit,
204        _inherited_implication: &impl Implication,
205        source: TransactionSource,
206    ) -> ValidateResult<Self::Val, RuntimeCallFor<Runtime>> {
207        // we only care about unsigned calls
208        if origin.as_system_origin_signer().is_some() {
209            return Ok((ValidTransaction::default(), (), origin));
210        };
211
212        let domains_call = match call.maybe_domains_call() {
213            Some(domains_call) => domains_call,
214            None => return Ok((ValidTransaction::default(), (), origin)),
215        };
216
217        let validity = match domains_call {
218            DomainsCall::submit_bundle { opaque_bundle } => {
219                Self::do_validate_submit_bundle(opaque_bundle, source)?
220            }
221            DomainsCall::submit_fraud_proof { fraud_proof } => {
222                Self::do_validate_fraud_proof(fraud_proof, source)?
223            }
224            DomainsCall::submit_receipt { singleton_receipt } => {
225                Self::do_validate_singleton_receipt(singleton_receipt, source)?
226            }
227            _ => return Err(InvalidTransaction::Call.into()),
228        };
229
230        Ok((validity, (), Origin::ValidatedUnsigned.into()))
231    }
232
233    impl_tx_ext_default!(RuntimeCallFor<Runtime>; prepare);
