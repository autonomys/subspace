30type OperatorOf<T> =
31    Operator<BalanceOf<T>, <T as Config>::Share, DomainBlockNumberFor<T>, ReceiptHashFor<T>>;

33/// Fetch the next epoch's operators from the DomainStakingSummary
34pub(crate) fn get_next_operators<T: Config>(domain_id: DomainId) -> Vec<OperatorOf<T>> {
35    let domain_summary = DomainStakingSummary::<T>::get(domain_id)
36        .expect("invariant violated: We must have DomainStakingSummary");
37    let mut prev_ops = vec![];
38    for operator_id in &domain_summary.next_operators {
39        let operator = Operators::<T>::get(*operator_id).expect(
40            "invariant violated: Operator in next_operator set is not present in Operators",
41        );
42        prev_ops.push(operator)
43    }
44    prev_ops
45}

47/// Finalize the epoch and transition to the next one
48pub(crate) fn conclude_domain_epoch<T: Config>(domain_id: DomainId) {
49    let head_domain_number = HeadDomainNumber::<T>::get(domain_id);
50    HeadDomainNumber::<T>::set(domain_id, head_domain_number + One::one());
51    do_finalize_domain_current_epoch::<T>(domain_id)
52        .expect("invariant violated: we must be able to finalize domain epoch");
53}

55/// Mark an operator as having produced an invalid bundle
56pub(crate) fn fuzz_mark_invalid_bundle_authors<T: Config<DomainHash = H256>>(
57    operator: OperatorId,
58    domain_id: DomainId,
59) -> Option<H256> {
60    let pending_slashes = PendingSlashes::<T>::get(domain_id).unwrap_or_default();
61    let mut invalid_bundle_authors_in_epoch = InvalidBundleAuthors::<T>::get(domain_id);
62    let mut stake_summary = DomainStakingSummary::<T>::get(domain_id).unwrap();
63    if pending_slashes.contains(&operator) {
64        return None;
65    }
66    let er = H256::random();
67    mark_invalid_bundle_author::<T>(
68        operator,
69        er,
70        &mut stake_summary,
71        &mut invalid_bundle_authors_in_epoch,
72    )
73    .expect("invariant violated: could not mark operator as invalid bundle author");
74    DomainStakingSummary::<T>::insert(domain_id, stake_summary);
75    InvalidBundleAuthors::<T>::insert(domain_id, invalid_bundle_authors_in_epoch);
76    Some(er)
77}

79/// Unmark an operator as having produced an invalid bundle
80pub(crate) fn fuzz_unmark_invalid_bundle_authors<T: Config<DomainHash = H256>>(
81    domain_id: DomainId,
82    operator: OperatorId,
83    er: H256,
84) {
85    let pending_slashes = PendingSlashes::<T>::get(domain_id).unwrap_or_default();
86    let mut invalid_bundle_authors_in_epoch = InvalidBundleAuthors::<T>::get(domain_id);
87    let mut stake_summary = DomainStakingSummary::<T>::get(domain_id).unwrap();
88
89    if pending_slashes.contains(&operator)
90        || crate::Pallet::<T>::is_operator_pending_to_slash(domain_id, operator)
91    {
92        return;
93    }
94
95    unmark_invalid_bundle_author::<T>(
96        operator,
97        er,
98        &mut stake_summary,
99        &mut invalid_bundle_authors_in_epoch,
100    )
101    .expect("invariant violated: could not unmark operator as invalid bundle author");
102
103    DomainStakingSummary::<T>::insert(domain_id, stake_summary);
104    InvalidBundleAuthors::<T>::insert(domain_id, invalid_bundle_authors_in_epoch);
105}

112/// Check staking invariants before epoch finalization
113pub(crate) fn check_invariants_before_finalization<T: Config>(domain_id: DomainId) {
114    let domain_summary = DomainStakingSummary::<T>::get(domain_id).unwrap();
115    // INVARIANT: all current_operators are registered and not slashed nor have invalid bundles
116    for operator_id in &domain_summary.next_operators {
117        let operator = Operators::<T>::get(*operator_id).unwrap();
118        if !matches!(
119            operator.status::<T>(*operator_id),
120            OperatorStatus::Registered
121        ) {
122            panic!("operator set violated");
123        }
124    }
125    // INVARIANT: No operator is common between DeactivatedOperator and DeregisteredOperator
126    let deactivated_operators = DeactivatedOperators::<T>::get(domain_id);
127    let deregistered_operators = DeregisteredOperators::<T>::get(domain_id);
128    for operator_id in &deregistered_operators {
129        assert!(!deactivated_operators.contains(operator_id));
130    }
131}

133/// Check staking invariants after epoch finalization
134pub(crate) fn check_invariants_after_finalization<T: Config<Balance = u128, Share = u128>>(
135    domain_id: DomainId,
136    prev_ops: Vec<OperatorOf<T>>,
137) {
138    let domain_summary = DomainStakingSummary::<T>::get(domain_id).unwrap();
139    for operator_id in domain_summary.current_operators.keys() {
140        let operator = Operators::<T>::get(operator_id).unwrap();
141        // INVARIANT: 0 < SharePrice < 1
142        SharePrice::new::<T>(operator.current_total_shares, operator.current_total_stake)
143            .expect("SharePrice to be present");
144    }
145
146    // INVARIANT: DeactivatedOperators is empty
147    let deactivated_operators = DeactivatedOperators::<T>::get(domain_id);
148    assert!(deactivated_operators.is_empty());
149    // INVARIANT: DeregisteredOperators is empty
150    let deregistered_operators = DeregisteredOperators::<T>::get(domain_id);
151    assert!(deregistered_operators.is_empty());
152
153    // INVARIANT: Total domain stake == accumulated operators' curent_stake.
154    let aggregated_stake: BalanceOf<T> = domain_summary
155        .current_operators
156        .values()
157        .fold(0, |acc, stake| acc.saturating_add(*stake));
158
159    assert_eq!(aggregated_stake, domain_summary.current_total_stake);
160    // INVARIANT: all current_operators are registered and not slashed nor have invalid bundles
161    for operator_id in domain_summary.current_operators.keys() {
162        let operator = Operators::<T>::get(operator_id).unwrap();
163        if !matches!(
164            operator.status::<T>(*operator_id),
165            OperatorStatus::Registered
166        ) {
167            panic!("operator set violated");
168        }
169        // INVARIANT: Shares add up
170        let mut shares: T::Share = 0;
171        for (operator, _nominator, deposit) in Deposits::<T>::iter() {
172            if *operator_id == operator {
173                shares += deposit.known.shares;
174            }
175        }
176        assert!(shares <= operator.current_total_shares);
177    }
178
179    // INVARIANT: all operators which were part of the next operator set before finalization are present now
180    assert_eq!(prev_ops.len(), domain_summary.current_operators.len());
181}

183/// Check general Substrate invariants that must always hold
184pub(crate) fn check_general_invariants<
185    T: Config<Balance = u128>
186        + pallet_balances::Config<Balance = u128>
187        + frame_system::Config<AccountData = pallet_balances::AccountData<u128>>,
188>(
189    initial_total_issuance: BalanceOf<T>,
190) {
191    // After execution of all blocks, we run invariants
192    let mut counted_free: <T as pallet_balances::Config>::Balance = 0;
193    let mut counted_reserved: <T as pallet_balances::Config>::Balance = 0;
194    for (account, info) in Account::<T>::iter() {
195        let consumers = info.consumers;
196        let providers = info.providers;
197        assert!(
198            !(consumers > 0 && providers == 0),
199            "Invalid account consumers or providers state"
200        );
201        counted_free += info.data.free;
202        counted_reserved += info.data.reserved;
203        let max_lock: <T as pallet_balances::Config>::Balance =
204            pallet_balances::Locks::<T>::get(&account)
205                .iter()
206                .map(|l| l.amount)
207                .max()
208                .unwrap_or_default();
209        assert_eq!(
210            max_lock, info.data.frozen,
211            "Max lock should be equal to frozen balance"
212        );
213        let sum_holds: <T as pallet_balances::Config>::Balance =
214            Holds::<T>::get(&account).iter().map(|l| l.amount).sum();
215        assert!(
216            sum_holds <= info.data.reserved,
217            "Sum of all holds ({sum_holds}) should be less than or equal to reserved balance {}",
218            info.data.reserved
219        );
220    }
221    let total_issuance = TotalIssuance::<T>::get();
222    let counted_issuance = counted_free + counted_reserved;
223    assert_eq!(total_issuance, counted_issuance);
224    assert!(total_issuance >= initial_total_issuance);
225}
