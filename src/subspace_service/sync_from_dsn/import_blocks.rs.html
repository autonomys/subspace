24/// How many blocks to queue before pausing and waiting for blocks to be imported, this is
25/// essentially used to ensure we use a bounded amount of RAM during sync process.
26const QUEUED_BLOCKS_LIMIT: BlockNumber = 500;
27/// Time to wait for blocks to import if import is too slow
28const WAIT_FOR_BLOCKS_TO_IMPORT: Duration = Duration::from_secs(1);
29
30/// Starts the process of importing blocks.
31///
32/// Returns number of downloaded blocks.
33#[allow(clippy::too_many_arguments)]
34pub(super) async fn import_blocks_from_dsn<Block, AS, Client, PG, IQS>(
35    segment_headers_store: &SegmentHeadersStore<AS>,
36    segment_header_downloader: &SegmentHeaderDownloader,
37    client: &Client,
38    piece_getter: &PG,
39    import_queue_service: &mut IQS,
40    last_completed_segment_index: &mut SegmentIndex,
41    last_processed_block_number: &mut NumberFor<Block>,
42    erasure_coding: &ErasureCoding,
43) -> Result<u64, Error>
44where
45    Block: BlockT,
46    AS: AuxStore + Send + Sync + 'static,
47    Client: HeaderBackend<Block> + BlockBackend<Block> + Send + Sync + 'static,
48    PG: PieceGetter,
49    IQS: ImportQueueService<Block> + ?Sized,
50{
51    {
52        let last_segment_header = segment_headers_store.last_segment_header().ok_or_else(|| {
53            Error::Other(
54                "Archiver needs to be initialized before syncing from DSN to populate the very \
55                first segment"
56                    .to_string(),
57            )
58        })?;
59
60        let new_segment_headers = segment_header_downloader
61            .get_segment_headers(&last_segment_header)
62            .await
63            .map_err(|error| error.to_string())?;
64
65        debug!("Found {} new segment headers", new_segment_headers.len());
66
67        if !new_segment_headers.is_empty() {
68            segment_headers_store.add_segment_headers(&new_segment_headers)?;
69        }
70    }
71
72    let mut imported_blocks = 0;
73    let mut reconstructor = Arc::new(Mutex::new(Reconstructor::new(erasure_coding.clone())));
74    // Start from the first unprocessed segment and process all segments known so far
75    let segment_indices_iter = (*last_completed_segment_index + SegmentIndex::ONE)
76        ..=segment_headers_store
77            .max_segment_index()
78            .expect("Exists, we have inserted segment headers above; qed");
79    let mut segment_indices_iter = segment_indices_iter.peekable();
80
81    while let Some(segment_index) = segment_indices_iter.next() {
82        debug!( %segment_index, "Processing segment");
83
84        let segment_header = segment_headers_store
85            .get_segment_header(segment_index)
86            .expect("Statically guaranteed to exist, see checks above; qed");
87
88        let last_archived_maybe_partial_block_number = segment_header.last_archived_block().number;
89        let last_archived_block_partial = segment_header
90            .last_archived_block()
91            .archived_progress
92            .partial()
93            .is_some();
94
95        trace!(
96            %segment_index,
97            last_archived_maybe_partial_block_number,
98            last_archived_block_partial,
99            "Checking segment header"
100        );
101
102        let info = client.info();
103        let last_archived_maybe_partial_block_number =
104            NumberFor::<Block>::from(last_archived_maybe_partial_block_number);
105        // We have already processed the last block in this segment, or one higher than it,
106        // so it can't change. Resetting the reconstructor loses any partial blocks, so we
107        // only reset if the (possibly partial) last block has been processed.
108        if *last_processed_block_number >= last_archived_maybe_partial_block_number {
109            debug!(
110                %segment_index,
111                %last_processed_block_number,
112                %last_archived_maybe_partial_block_number,
113                %last_archived_block_partial,
114                "Already processed last (possibly partial) block in segment, resetting reconstructor",
115            );
116            *last_completed_segment_index = segment_index;
117            // Reset reconstructor instance
118            reconstructor = Arc::new(Mutex::new(Reconstructor::new(erasure_coding.clone())));
119            continue;
120        }
121        // Just one partial unprocessed block and this was the last segment available, so nothing to
122        // import. (But we also haven't finished this segment yet, because of the partial block.)
123        if last_archived_maybe_partial_block_number == *last_processed_block_number + One::one()
124            && last_archived_block_partial
125        {
126            if segment_indices_iter.peek().is_none() {
127                // We haven't fully processed this segment yet, because it ends with a partial block.
128                *last_completed_segment_index = segment_index.saturating_sub(SegmentIndex::ONE);
129
130                // We don't need to reset the reconstructor here. We've finished getting blocks, so
131                // we're about to return and drop the reconstructor and its partial block anyway.
132                // (Normally, we'd need that partial block to avoid a block gap. But we should be close
133                // enough to the tip that normal syncing will fill any gaps.)
134                debug!(
135                    %segment_index,
136                    %last_processed_block_number,
137                    %last_archived_maybe_partial_block_number,
138                    %last_archived_block_partial,
139                    "No more segments, snap sync is about to finish",
140                );
141                continue;
142            } else {
143                // Downloading an entire segment for one partial block should be rare, but if it
144                // happens a lot we want to see it in the logs.
145                //
146                // TODO: if this happens a lot, check for network/DSN sync bugs - we should be able
147                // to sync to near the tip reliably, so we don't have to keep reconstructor state.
148                info!(
149                    %segment_index,
150                    %last_processed_block_number,
151                    %last_archived_maybe_partial_block_number,
152                    %last_archived_block_partial,
153                    "Downloading entire segment for one partial block",
154                );
155            }
156        }
157
158        let segment_pieces = download_segment_pieces(
159            segment_index,
160            piece_getter,
161            SEGMENT_DOWNLOAD_RETRIES,
162            Some(SEGMENT_DOWNLOAD_RETRY_DELAY),
163        )
164        .await
165        .map_err(|error| {
166            format!("Failed to download segment pieces during block import: {error}")
167        })?;
168        // CPU-intensive piece and segment reconstruction code can block the async executor.
169        let segment_contents_fut = task::spawn_blocking({
170            let reconstructor = reconstructor.clone();
171
172            move || {
173                reconstructor
174                    .lock()
175                    .expect("Panic if previous thread panicked when holding the mutex")
176                    .add_segment(segment_pieces.as_ref())
177            }
178        });
179        let blocks = segment_contents_fut
180            .await
181            .expect("Panic if blocking task panicked")
182            .map_err(|error| error.to_string())?
183            .blocks;
184        trace!( %segment_index, "Segment reconstructed successfully");
185
186        let mut blocks_to_import = Vec::with_capacity(QUEUED_BLOCKS_LIMIT as usize);
187
188        let mut best_block_number = info.best_number;
189        for (block_number, block_bytes) in blocks {
190            let block_number = block_number.into();
191            if block_number == 0u32.into() {
192                let signed_block = client
193                    .block(
194                        client
195                            .hash(block_number)?
196                            .expect("Genesis block hash must always be found; qed"),
197                    )?
198                    .expect("Genesis block data must always be found; qed");
199
200                if encode_block(signed_block) != block_bytes {
201                    return Err(Error::Other(
202                        "Wrong genesis block, block import failed".to_string(),
203                    ));
204                }
205            }
206
207            // Limit number of queued blocks for import
208            // NOTE: Since best block number might be non-canonical, we might actually have more
209            // than `QUEUED_BLOCKS_LIMIT` elements in the queue, but it should be rare and
210            // insignificant. Feel free to address this in case you have a good strategy, but it
211            // seems like complexity is not worth it.
212            while block_number.saturating_sub(best_block_number) >= QUEUED_BLOCKS_LIMIT.into() {
213                let just_queued_blocks_count = blocks_to_import.len();
214                if !blocks_to_import.is_empty() {
215                    // This vector is quite large (~150kB), so replacing it with an uninitialized
216                    // vector with the correct capacity is faster than cloning and clearing it.
217                    // (Cloning requires a memcpy, which pages in and sets all the memory, which is
218                    // a waste just before clearing it.)
219                    let importing_blocks = std::mem::replace(
220                        &mut blocks_to_import,
221                        Vec::with_capacity(QUEUED_BLOCKS_LIMIT as usize),
222                    );
223                    // Import queue handles verification and importing it into the client
224                    import_queue_service
225                        .import_blocks(BlockOrigin::NetworkInitialSync, importing_blocks);
226                }
227                trace!(
228                    %block_number,
229                    %best_block_number,
230                    %just_queued_blocks_count,
231                    %QUEUED_BLOCKS_LIMIT,
232                    "Number of importing blocks reached queue limit, waiting before retrying"
233                );
234                tokio::time::sleep(WAIT_FOR_BLOCKS_TO_IMPORT).await;
235                best_block_number = client.info().best_number;
236            }
237
238            let signed_block =
239                decode_block::<Block>(&block_bytes).map_err(|error| error.to_string())?;
240
241            *last_processed_block_number = block_number;
242
243            // No need to import blocks that are already present, if block is not present it might
244            // correspond to a short fork, so we need to import it even if we already have another
245            // block at this height
246            if client.expect_header(signed_block.block.hash()).is_ok() {
247                continue;
248            }
249
250            let SignedBlock {
251                block,
252                justifications,
253            } = signed_block;
254            let (header, extrinsics) = block.deconstruct();
255            let hash = header.hash();
256
257            blocks_to_import.push(IncomingBlock {
258                hash,
259                header: Some(header),
260                body: Some(extrinsics),
261                indexed_body: None,
262                justifications,
263                origin: None,
264                allow_missing_state: false,
265                import_existing: false,
266                state: None,
267                skip_execution: false,
268            });
269
270            imported_blocks += 1;
271
272            if imported_blocks % 1000 == 0 {
273                debug!("Adding block {} from DSN to the import queue", block_number);
274            }
275        }
276
277        if !blocks_to_import.is_empty() {
278            // Import queue handles verification and importing it into the client
279            import_queue_service.import_blocks(BlockOrigin::NetworkInitialSync, blocks_to_import);
280        }
281
282        // Segments are only fully processed when all their blocks are fully processed.
283        if last_archived_block_partial {
284            *last_completed_segment_index = segment_index.saturating_sub(SegmentIndex::ONE);
285        } else {
286            *last_completed_segment_index = segment_index;
287        }
288    }
289
290    Ok(imported_blocks)
291}
